function addScript( url, callback ) {
    var script = document.createElement( 'script' );
    if( callback ) script.onload = callback;
    script.type = 'text/javascript';
    script.src = url;
    document.body.appendChild( script );

function createCookieDetails(name, value, hour){
	  var expires = "";
	  if (hour) {
	    var date = new Date();
	    date.setTime(date.getTime()+(hour*60*60*1000));
	    expires ="; expires="+date.toGMTString();
	  }
	  document.cookie = name+"="+value+expires+"; domain=.magicbricks.com; path=/";
}

function readCookie(name){
	  var ca = document.cookie.split(';');
	  var nameEQ = name + "=";
	  for(var i=0; i < ca.length; i++) {
	    var c = ca[i];
	    while (c.charAt(0)==' ') c = c.substring(1, c.length); 
	    if (c.indexOf(nameEQ) == 0){
	    	retVal=c.substring(nameEQ.length, c.length);
	    	if(name=='userTypeflSlider' && retVal.indexOf(",")>0)
	    		return retVal.substring(0, retVal.indexOf(","));
	    	else if(name=='userEmailflSlider'){
		    	return retVal.replace(/^"(.*)"$/, '$1');
		    }else if(name=='userNameflSlider'){
		    	return retVal.replace(/^"(.\*)"$/, '$1');
		    }
	    	else
	    		return retVal;
	    }
	   }
	  return null;
}

	  var expires = "";

	    var date = new Date();

	    var c = ca[i];
	    while (c.charAt(0)==' ') c = c.substring(1, c.length);
