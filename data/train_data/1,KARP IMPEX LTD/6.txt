	var globalNavMenuTimer = new Array();
	function globalNavMenuOnMouseOut(id) {
		globalNavMenuTimer[id] = setTimeout( function(){globalNavMenuCloseGlobalNavMenu(id);},100);
	}
	function globalNavMenuOnMouseOver(id) {
		clearTimeout(globalNavMenuTimer[id]);
	}
	function globalNavMenuCloseGlobalNavMenu(id) {
		clearTimeout(globalNavMenuTimer[id]);
		var divs = document.getElementById(id).getElementsByTagName("div");
		for (var i=0; i<divs.length; i++) {
			if (divs[i].id.indexOf("submenu")==0) {
				if ( divs[i].className.lastIndexOf("_shown")>=0 ) {
					divs[i].className = divs[i].className.substring(0, divs[i].className.lastIndexOf("_shown"));
				}
			}
		}
		document.getElementById(id).className = "";
	}
	function globalNavMenuOnClick(id) {
		var divs = document.getElementById(id).getElementsByTagName("div");
		for (var i=0; i<divs.length; i++) {
			if (divs[i].id.indexOf("submenu")==0) {
				if ( divs[i].className.indexOf("_shown")<0 ) {
					divs[i].className += "_shown";
				}
			}
		}
		document.getElementById(id).className = "menu_shown";
	}
	function updateGlobalNavMenu(){
		var objects = document.getElementsByTagName("li");
		for (var i=0; i<objects.length; i++) {
			if (objects[i].id.indexOf("gnavmenu")==0) {
				if (objects[i].addEventListener != null) {
					objects[i].addEventListener('mouseover',function(evt){globalNavMenuOnMouseOver(this.id);},false);
					objects[i].addEventListener('mouseout',function(evt){globalNavMenuOnMouseOut(this.id);},false);
					objects[i].addEventListener('click',function(evt){globalNavMenuOnClick(this.id);evt.preventDefault();},false);
					
				} else {
					objects[i].onmouseover = new Function ("evt", "globalNavMenuOnMouseOver('"+objects[i].id+"');");
					objects[i].onmouseout = new Function ("evt", "globalNavMenuOnMouseOut('"+objects[i].id+"');");
					objects[i].onclick = new Function ("evt", "globalNavMenuOnClick('"+objects[i].id+"');return false;");
					
				}
				var linkObjs = document.getElementById(objects[i].id).getElementsByTagName("a");
				for (var j=0; j<linkObjs.length; j++) {
					if (linkObjs[j].className!="drop") {
						var link = "";
						try {
							link = decodeURIComponent(escape(linkObjs[j].getAttribute("href")));
						} catch(err) { }
						if ( link.length>0 && link.indexOf("#")<0 ) {
							if (linkObjs[j].getAttribute("target")=="_blank"){
								linkObjs[j].onclick = new Function ("evt", "window.open('"+link+"')");
							} else {
								linkObjs[j].onclick = new Function ("evt", "window.location.href='"+link+"'");
							}
						}
					}
				}
			}
		}
	}
	
	function highlightFirstLevel(){
		var objects = document.getElementsByTagName("li");
		for (var i=0; i<objects.length; i++) {
			if (objects[i].id.indexOf("gnavmenu")==0) {
				var anchor_link = objects[i].getElementsByTagName("a");
				for (var j=0; j < anchor_link.length; j++){
					var str = document.location.href; //Current URL
					var temp = str.split("/");
					var pattern = anchor_link[j].href; //Hyperlink of menu item
					if (str.match(pattern) || anchor_link[j].id == temp[3]){ 
						var object_items = objects[i].children;
						object_items[0].className +=" current";
						return;
					}
				}
			}
		}
		
		for (var i=0;i<objects.length; i++){ //If it's unable to search for a menu item for highlight, highlight a default menu item
			if (objects[i].id.indexOf("gnavmenu")==0) { 
				var object_items = objects[i].children;
				if (object_items[0].id == "default"){
					object_items[0].className +=" current";
					}
			}
		}
	}

     function gotoSearchPage() {
     	var kword = document.getElementById("searchValueBox").value;
     	if (kword && kword.length > 0) {
     		var regex=/[\s&\(\)\/'\-%~!@#$^*_+=|{}\[\]\":\;`<>,.?\\]/gi;
     		var regex2=/[<>]/gi;
     		var alphanumeric = /^[a-zA-Z0-9]$/gi;
     		
			var str = kword.replace(regex, '');
     		if (str.length == 0){
     			alert('Please enter a valid searching criteria!');
     		} else if (str.match(alphanumeric)) {
     			alert('Please input a keyword with at least two characters.');
     		} else {
				kword = kword.replace(regex2,'');

		}

	   		}
	    }

	function checkDup(keywordCookie, kword, delimiter) {
		if(delimiter == null) {
			delimiter = ',';
		}
		var newKeywords = new Array();
		var oldKeywords = new Array();
		if (keywordCookie){
			oldKeywords = keywordCookie.split(delimiter);
		}
		var index = 0;
		//Deduplicate
		for (var i = 0; i < oldKeywords.length; i++) {
			if (oldKeywords[i] != kword) {
				newKeywords[index++] = oldKeywords[i];
			}
		}
		newKeywords[index++] = kword;
		
		//Only store 3 keywords in the cookie
		if (newKeywords.length > CAT_MAX_COOKIE_COUNT){
			newKeywords.splice(0, newKeywords.length - CAT_MAX_COOKIE_COUNT);
		}
		return newKeywords.join(delimiter);
	}

	function delegateMouseEvent(id, event) {
		if(document.createEvent) {
			var ce = document.createEvent("MouseEvents"); 
			if (event == "click") {
				ce.initEvent("click", true, false); 
			} else if (event == "mouseover") {
				ce.initEvent("mouseover", true, false); 
			} else if (event == "mouseout") {
				ce.initEvent("mouseout", true, false); 
			}
			document.getElementById(id).dispatchEvent(ce); 
		} else { 
			if (event == "click") {
				document.getElementById(id).click();         
			} else if (event == "mouseover") {
				document.getElementById(id).onmouseover();         
			} else if (event == "mouseout") {
				document.getElementById(id).onmouseout();
			}
		} 
	}

     function gotoSearchPage() {
     	var kword = document.getElementById("searchValueBox").value;
     	if (kword && kword.length > 0) {
     		var regex=/[\s&\(\)\/'\-%~!@#$^*_+=|{}\[\]\":\;`<>,.?\\]/gi;
     		var regex2=/[<>]/gi;
     		var alphanumeric = /^[a-zA-Z0-9]$/gi;
     		
			var str = kword.replace(regex, '');
     		if (str.length == 0){
     			alert('Please enter a valid searching criteria!');
     		} else if (str.match(alphanumeric)) {
     			alert('Please input a keyword with at least two characters.');
     		} else {
				kword = kword.replace(regex2,'');
				return gotoSearchPageWithKeyWord(kword);
		    }
		}
	    else {
	    	var tagSelected=document.getElementById(formName + ":searchTabHidden").value;
	   		
	   		if (tagSelected == "tagProduct") {
	   			var selectedPathId = document.getElementById("pathId").value;
	   			if (selectedPathId == -1){
	   				alert('Please enter a keyword!');
	   			} else {
	   				var searchPageLink = categoryMapping[selectedPathId];
	   				if (searchPageLink){
	   					window.location = searchPageLink;
	   				}
	   			}
	   		} else {
	   			alert('Please enter a keyword!');
	   		}
	    }
     }
     
	function checkDup(keywordCookie, kword, delimiter) {
		if(delimiter == null) {
			delimiter = ',';
		}
		var newKeywords = new Array();
		var oldKeywords = new Array();
		if (keywordCookie){
			oldKeywords = keywordCookie.split(delimiter);
		}
		var index = 0;
		//Deduplicate
		for (var i = 0; i < oldKeywords.length; i++) {
			if (oldKeywords[i] != kword) {
				newKeywords[index++] = oldKeywords[i];
			}
		}
		newKeywords[index++] = kword;
		
		//Only store 3 keywords in the cookie
		if (newKeywords.length > CAT_MAX_COOKIE_COUNT){
			newKeywords.splice(0, newKeywords.length - CAT_MAX_COOKIE_COUNT);
		}
		return newKeywords.join(delimiter);
	}

	function delegateMouseEvent(id, event) {
		if(document.createEvent) {
			var ce = document.createEvent("MouseEvents"); 
			if (event == "click") {
				ce.initEvent("click", true, false); 
			} else if (event == "mouseover") {
				ce.initEvent("mouseover", true, false); 
			} else if (event == "mouseout") {
				ce.initEvent("mouseout", true, false); 
			}
			document.getElementById(id).dispatchEvent(ce); 
		} else { 
			if (event == "click") {
				document.getElementById(id).click();         
			} else if (event == "mouseover") {
				document.getElementById(id).onmouseover();         
			} else if (event == "mouseout") {
				document.getElementById(id).onmouseout();
			}
		} 
	}
