	function get_final_rating(ratings){
		var one_rating_count = 		parseInt(ratings[0]['1']);
		var two_rating_count = 		parseInt(ratings[0]['2']);
		var three_rating_count = 	parseInt(ratings[0]['3']);
		var four_rating_count = 	parseInt(ratings[0]['4']);
		var five_rating_count = 	parseInt(ratings[0]['5']);
		
		var total_rating_count = one_rating_count+two_rating_count+three_rating_count+four_rating_count+five_rating_count;
		var sum_rating = (one_rating_count*1)+(two_rating_count*2)+(three_rating_count*3)+(four_rating_count*4)+(five_rating_count*5);
		
		final_rating = parseInt(sum_rating / total_rating_count);
		return final_rating;
	}
	
	function checkCookie(postid) {
		var user= getCookie("ratings");
		if (user != "") {
			//alert("Welcome again " + user);
			return 1;
		} else {
		   //user = prompt("Please enter your name:","");
		   user = postid;
		   if(user != "" && user != null) {
			   setCookie("ratings", postid, 999);
			   return 0;
		   }
		}
	}

	function set_stars(therate){
		for(var i = 1; i <= therate; i++) {
			$('.rate-btn-'+i).addClass('rate-btn-hover');
		}
	}

		var three_rating_count = 	parseInt(ratings[0]['3']);

		var five_rating_count = 	parseInt(ratings[0]['5']);
		
		var total_rating_count = one_rating_count+two_rating_count+three_rating_count+four_rating_count+five_rating_count;
		var sum_rating = (one_rating_count*1)+(two_rating_count*2)+(three_rating_count*3)+(four_rating_count*4)+(five_rating_count*5);
