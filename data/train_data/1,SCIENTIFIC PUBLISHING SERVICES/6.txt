    function request_invite() {
        if ($('#invitationBox').val().trim() == '' || $('#invitationBox').val().trim() == 'Enter your email address') {
            return false;
        }
        $.ajax({
            url: '/EmailSignUpFormPart.aspx',
            data: { email: $('#invitationBox').val().trim() },
            success: function (data) {
                if (data.status == "success") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#Success').fadeIn('fast');
                    $('#RequestInvitation').hide();
                } else if (data.status == "alreadyLoggedIn") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#alreadyLoggedIn').fadeIn('fast');
                    $('#RequestInvitation').hide();
                }
                else if (data.status == "emailAlreadyExists") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#emailAlreadyExists').fadeIn('fast');
                    $('#RequestInvitation').hide();
                }
                else {
                    $('#CompleteAccount .BoardNameError').html('<span style="color:red;">' + data.message + '</span>').show();
                }
            },
            error: function (response) {
                console.log(response);
            }
        });

        window.dataLayer = window.dataLayer || [];
        window.dataLayer.push({ 'event': 'interactiveEvent', 'eventCategory': 'RC Email Sign Up', 'eventAction': 'RC Email Sign Up' });

        return true;
    }

    function request_invite() {
        if ($('#invitationBox').val().trim() == '' || $('#invitationBox').val().trim() == 'Enter your email address') {
            return false;
        }
        $.ajax({
            url: '/EmailSignUpFormPart.aspx',
            data: { email: $('#invitationBox').val().trim() },
            success: function (data) {
                if (data.status == "success") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#Success').fadeIn('fast');
                    $('#RequestInvitation').hide();
                } else if (data.status == "alreadyLoggedIn") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#alreadyLoggedIn').fadeIn('fast');
                    $('#RequestInvitation').hide();
                }
                else if (data.status == "emailAlreadyExists") {
                    $('#invitationBox').val('Email');
                    $('#invitationBox').blur();
                    $('#emailAlreadyExists').fadeIn('fast');
                    $('#RequestInvitation').hide();
                }
                else {
                    $('#CompleteAccount .BoardNameError').html('<span style="color:red;">' + data.message + '</span>').show();
                }
            },
            error: function (response) {
                console.log(response);
            }
        });

        window.dataLayer = window.dataLayer || [];
        window.dataLayer.push({ 'event': 'interactiveEvent', 'eventCategory': 'RC Email Sign Up', 'eventAction': 'RC Email Sign Up' });

        return true;
    }

    $(document).ready(function () {
        $("#invitationBox").bind("keydown", function (e) {
            if (e.keyCode == 13) { // enter key
                request_invite();
                return false;
            }
            return true;
        });
    });

    function PSAddListener(el, sType, fn) {
        if (window.addEventListener) {
            el.addEventListener(sType, fn, false);
        }
        else if (window.attachEvent) {
            el.attachEvent("on" + sType, fn);
        }
    }

    function summaryMouseOver(e) {
        var el;
        if (!e) var e = window.event;
        if (e.target) el = e.target;
        else if (e.srcElement) el = e.srcElement;
        if (el.nodeType == 3)
            el = el.parentNode;

        el.style.cursor = 'pointer';
        createPopup('A1FBEA27-592D-4868-B89F-7DFA4808CB90', 'Click chart to get more specific', 'auto', 'B9D5E3', 'FFFFFF', true);
    }

    function summaryMouseOut(e) {
        var el;
        if (!e) var e = window.event;
        if (e.target) el = e.target;
        else if (e.srcElement) el = e.srcElement;
        if (el.nodeType == 3)
            el = el.parentNode;

        el.style.cursor = '';
        hidePopup('A1FBEA27-592D-4868-B89F-7DFA4808CB90');
    }

    function summaryClick(e) {
        startWizard();
