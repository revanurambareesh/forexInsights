        var SelectedDiv = "";
        function OnClickEvents(ctlid, ctrl) {
            if (SelectedDiv != ctrl) {
                $('.cls-CommingSoon').removeClass('cls-CommingSoon-selected');
                $('#' + ctrl).addClass('cls-CommingSoon-selected');
                $('.cls-container').removeClass('cls-show-Container').addClass('cls-hide-Container');
                $('#divContainer_' + ctlid).removeClass('cls-hide-Container').addClass("cls-show-Container");
                if (ctlid == "1") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "block");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#dvMyCat').html("About Us");
                }
                else if (ctlid == "2") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "block");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#dvMyCat').html("Media Room");

                }
                else if (ctlid == "3") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "block");
                    $('#dvMyCat').html("Investor Relations");

                }
                else if (ctlid == "4") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvCompliances').css("display", "block");
                    $('#dvMyCat').html("Compliances");

                }
    SelectedDiv = ctrl;
}
    return false;
}

        var SelectedDiv = "";
        function OnClickEvents(ctlid, ctrl) {
            if (SelectedDiv != ctrl) {
                $('.cls-CommingSoon').removeClass('cls-CommingSoon-selected');
                $('#' + ctrl).addClass('cls-CommingSoon-selected');
                $('.cls-container').removeClass('cls-show-Container').addClass('cls-hide-Container');
                $('#divContainer_' + ctlid).removeClass('cls-hide-Container').addClass("cls-show-Container");
                if (ctlid == "1") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "block");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#dvMyCat').html("About Us");
                }
                else if (ctlid == "2") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "block");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#dvMyCat').html("Media Room");

                }
                else if (ctlid == "3") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "block");
                    $('#dvMyCat').html("Investor Relations");

                }
                else if (ctlid == "4") {
                    $('#ctl00_ContentPlaceHolder1_dvAboutUsSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvMediaRoomSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvInvestorSection').css("display", "none");
                    $('#ctl00_ContentPlaceHolder1_dvCompliances').css("display", "block");
                    $('#dvMyCat').html("Compliances");

                }
    SelectedDiv = ctrl;
}
    return false;
}
