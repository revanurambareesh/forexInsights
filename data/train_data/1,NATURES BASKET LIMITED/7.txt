        function getdaysfromnumber(days) {
            var daysarray = days.split(',');
            var count = daysarray.length - 1;
            var daysstring = '';
            for (var i in daysarray) {
                if (count == i) {
                    daysstring = daysstring.slice(0, -1);
                    if (count >= 1)
                        daysstring += ' and ';
                    daysstring += daysarray[i];
                }
                else {
                    daysstring += ' ' + daysarray[i] + ',';
                }
            }
            daysstring = daysstring.replace('1', 'Sunday').replace('2', 'Monday').replace('3', 'Tuesday').replace('4', 'Wednesday').replace('5', 'Thursday').replace('6', 'Friday').replace('7', 'Saturday')
            return daysstring;
        }

        function getdaysfromnumber(days) {
            var daysarray = days.split(',');
            var count = daysarray.length - 1;
            var daysstring = '';
            for (var i in daysarray) {
                if (count == i) {
                    daysstring = daysstring.slice(0, -1);
                    if (count >= 1)
                        daysstring += ' and ';
                    daysstring += daysarray[i];
                }
                else {
                    daysstring += ' ' + daysarray[i] + ',';
                }
            }
            daysstring = daysstring.replace('1', 'Sunday').replace('2', 'Monday').replace('3', 'Tuesday').replace('4', 'Wednesday').replace('5', 'Thursday').replace('6', 'Friday').replace('7', 'Saturday')
            return daysstring;
        }
