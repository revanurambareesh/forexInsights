function  footer_logurl(obj)
{
	if(typeof(glb_currentpage) != "undefined" )
	{
		var c_href = obj.href ;
		if(glb_currentpage == "home")
		{
			if(c_href.indexOf('?') == -1 )
			{
				document.location.href = c_href + "?src=all_pg";
			}
			else
			{
				document.location.href = c_href + "&src=all_pg";
			}
			return false;
		}
		else if(glb_currentpage == "company")
		{
			if(c_href.indexOf('?') == -1 )
			{
				document.location.href = c_href + "?src=comp_all_pg";
			}
			else
			{
				document.location.href = c_href + "&src=comp_all_pg";
			}
			return false;
		}
		else
		{
			return true;
		}
	}
	else
	{
		return true;
	}
}
