$(document).ready(function() {
	$("#submit_footer1").click(function(){
		var category1 = $('#category1').val();
		if(category1 == ''){
			alert('Please Select A Category');
			return false;
		}
	});
	$('#category1').change(function(){
		$(this).after('<div id="loader" class="col-sm-12"><img width="100%" height="15px" src="admin/images/loading.gif" alt="loading subcategory" /></div>');		
		var category1 = $('#category1 option:selected').val();
		$.ajax({
			url: "footer_ajax1.php",
			data: "category="+category1,
			type: "POST",
			cache: false,
			success : function(response){
				$('#type_response1').html(response);
				$('#loader').slideUp(200, function() {
					$(this).remove();
				});
			}
		});
	});
});
