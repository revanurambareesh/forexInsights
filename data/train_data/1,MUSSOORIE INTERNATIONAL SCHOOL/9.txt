  var vars = query.split("&");
  for (var i=0;i<vars.length;i++) {
    var pair = vars[i].split("=");
    if(pair.length > 0)
       if (pair[0] == variable) {
          return pair[1];
       }
  } 
  console.log('Query Variable ' + variable + ' not found');
  return "";
}

            c = c.substring(1);
        }
        if (c.indexOf(name) == 0) {
            return c.substring(name.length, c.length);
        }
    }
    return "";
}
