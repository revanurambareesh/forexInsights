				start: $scope.start,
				desiredNews: desiredhomeNews,
				filters : $scope.filter

					// Update the total number of items
					$scope.start += data.length;

					$scope.list = $scope.list.concat(data);
					// Make sure we execute the following code after the DOM
					// is succesfully loaded using $timeout
					$timeout(function() {
						// Get the ID of the first Item to load
						var firstAddedItemID = '#home-item-' + $scope.list[$scope.list.length - desiredNews]['ID'];

						var $widgetContent = document.querySelector('#HomeNewsCtrl .content');

						$widgetContent.scrollTop = $firstAddedItem.offsetTop - ($widgetContent.offsetTop + 15);
						
						$("#HomeNewsCtrl #spinner").hide();

				}
