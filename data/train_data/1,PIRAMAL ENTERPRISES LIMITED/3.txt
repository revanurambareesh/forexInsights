		function setOpacity()
		{ // Function for Timeline on Our History Page

					$('.hisLeft > ul').children('li').eq(leftCount).addClass('opaque_active');

					$('.hisRight > ul').children('li').eq(rightCount).addClass('opaque_active');

					leftBool = true;
					rightCount++;
					count++;
