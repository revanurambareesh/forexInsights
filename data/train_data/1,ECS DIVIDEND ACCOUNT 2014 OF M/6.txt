    function Workflow(folioNo, serType, msgType, msgDesc, PAN) {
        $.ajax({
            type: "POST",
            async: false,
            url: '/_vti_bin/SBIMF/SBIMFTrnsService.svc/WorkflowPreloginData',
            data: JSON.stringify({ folioNum: folioNo, ser_type: serType, msgType: msgType, msgDesc: msgDesc, PAN: PAN }),
            contentType: 'application/json; charset=utf-8',
            dataType: 'json',
            success: function (resp) {
                if (resp != "") {

                    idealake_website_loaderManagement.LoaderEnd();

                    $("#result").html("");
                    if (PAN== "") {

                        var response = JSON.parse(resp);
                        var table = "<table class='table-responsive'>";
                        table += "<th>Folio Number</th><th>Message</th>";
                        for (var i = 0; i < response.length; i++) {
                            var tr = "<tr>";
                            var td1 = "<td>" + response[i]["FolioNo"] + "</td>";
                            var td2 = "<td>" + response[i]["Return_Message"] + "</td></tr>";
                            table += (tr + td1 + td2);
                        }
                        table += "</table>";
                        $("#result").append(table);
                        addPagination(".table-responsive", 10);

                    }
                    else {

                        $("#result").html("<h5>Your request has been received, you will shortly recieve a mail.</h5>");
                    }


                }
                else {
                    alert('Request couldnt be generated. Please login.')
                }
            }, beforeSend: function () {
                idealake_website_loaderManagement.init();
            },
            complete: function () {
                idealake_website_loaderManagement.LoaderEnd();
            },
            error: function (xhr, status, e) {
                alert("There seems to be network issue.");
                idealake_website_loaderManagement.LoaderEnd();
            }
        });

    }
    function ValidateRegex(regex, inputValue) {
        if (regex != "") {
            return regex.test(inputValue);
        } else {
            return false;
        }

    }

    function ValidateElementByType(inputValue, inputType) {

        var result = false;
        var regex = "";
        var value = inputValue;
        var type = inputType;

        if (value != "" && type != "") {
            switch (type) {
                case "PAN":
                    regex = /^([a-zA-Z]){5}([0-9]){4}([a-zA-Z]){1}?$/;
                    //regex = /^([A-Z]{5}\d{4}[A-Z]){1}?$/;
                    break;

                case "number":
                    regex = /^[0-9]+$/;
                    break;

                default:
                    regex = ""
                    break;
            }
        }
        result = ValidateRegex(regex, value)
        return result
    }

    function InitializeNumbersOnly() {
        if ($('.numbers').length > 0) {
            $('.numbers').each(function (index) {
                $(this).keydown(function (e) {
                    if (e.shiftKey || e.ctrlKey || e.altKey) { // if shift, ctrl or alt keys held down
                        e.preventDefault();
                    }
                    else {
                        var key = e.charCode || e.keyCode || 0;
                        return (
                            key == 8 || //backspace
                            key == 9 || //tab
                            key == 46 || //delete

                        key == 110 || //decimal point
                        key == 190 || //Period
                        (key >= 35 && key <= 40) ||
                            //end  35  
                            //home  36  
                            //left arrow  37  
                            //up arrow  38  
                            //right arrow  39  
                            //down arrow  40  
                        (key >= 48 && key <= 57) ||    //num from 0-9
                        (key >= 96 && key <= 105));//num from 0-9
                    }
                });

            });
        };

    }

    function Workflow(folioNo, serType, msgType, msgDesc, PAN) {
        $.ajax({
            type: "POST",
            async: false,
            url: '/_vti_bin/SBIMF/SBIMFTrnsService.svc/WorkflowPreloginData',
            data: JSON.stringify({ folioNum: folioNo, ser_type: serType, msgType: msgType, msgDesc: msgDesc, PAN: PAN }),
            contentType: 'application/json; charset=utf-8',
            dataType: 'json',
            success: function (resp) {
                if (resp != "") {

                    idealake_website_loaderManagement.LoaderEnd();

                    $("#result").html("");
                    if (PAN== "") {

                        var response = JSON.parse(resp);
                        var table = "<table class='table-responsive'>";
                        table += "<th>Folio Number</th><th>Message</th>";
                        for (var i = 0; i < response.length; i++) {
                            var tr = "<tr>";
                            var td1 = "<td>" + response[i]["FolioNo"] + "</td>";
                            var td2 = "<td>" + response[i]["Return_Message"] + "</td></tr>";
                            table += (tr + td1 + td2);
                        }
                        table += "</table>";
                        $("#result").append(table);
                        addPagination(".table-responsive", 10);

                    }
                    else {

                        $("#result").html("<h5>Your request has been received, you will shortly recieve a mail.</h5>");
                    }


                }
                else {
                    alert('Request couldnt be generated. Please login.')
                }
            }, beforeSend: function () {
                idealake_website_loaderManagement.init();
            },
            complete: function () {
                idealake_website_loaderManagement.LoaderEnd();
            },
            error: function (xhr, status, e) {
                alert("There seems to be network issue.");
                idealake_website_loaderManagement.LoaderEnd();
            }
        });

    }
    function ValidateRegex(regex, inputValue) {
        if (regex != "") {
            return regex.test(inputValue);
        } else {
            return false;
        }

    }

    function ValidateElementByType(inputValue, inputType) {

        var result = false;
        var regex = "";
        var value = inputValue;
        var type = inputType;

        if (value != "" && type != "") {
            switch (type) {
                case "PAN":
                    regex = /^([a-zA-Z]){5}([0-9]){4}([a-zA-Z]){1}?$/;
                    //regex = /^([A-Z]{5}\d{4}[A-Z]){1}?$/;
                    break;

                case "number":
                    regex = /^[0-9]+$/;
                    break;

                default:
                    regex = ""
                    break;
            }
        }
        result = ValidateRegex(regex, value)
        return result
    }

    function InitializeNumbersOnly() {
        if ($('.numbers').length > 0) {
            $('.numbers').each(function (index) {
                $(this).keydown(function (e) {
                    if (e.shiftKey || e.ctrlKey || e.altKey) { // if shift, ctrl or alt keys held down
                        e.preventDefault();
                    }
                    else {
                        var key = e.charCode || e.keyCode || 0;
                        return (
                            key == 8 || //backspace
                            key == 9 || //tab
                            key == 46 || //delete

                        key == 110 || //decimal point
                        key == 190 || //Period
                        (key >= 35 && key <= 40) ||
                            //end  35  
                            //home  36  
                            //left arrow  37  
                            //up arrow  38  
                            //right arrow  39  
                            //down arrow  40  
                        (key >= 48 && key <= 57) ||    //num from 0-9
                        (key >= 96 && key <= 105));//num from 0-9
                    }
                });

            });
        };

    }
