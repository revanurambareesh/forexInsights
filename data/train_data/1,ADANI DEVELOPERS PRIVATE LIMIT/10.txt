	function getOtherAgents(){
		var devId = '55237';
		var cities = [] ;
		$("#otherCity:checked").each(function() {
	        cities.push($(this).val());
	    });
	    if(cities.length == 0){
	    	alert('Please select at least one city');
	    }else{
	    	$("#multiCityDD").hide();
	    	ajaxService.getOtherAgents(devId, cities, displayingCity, {callback:function(data){
	    		if(data){
	    			$("#normAgentSection").css("display", "none");
	    			$(".normAgentSection").css("display", "none");
	    			$("#otherCitiesOptionId").css("display", "none");
	    			$("#ajaxAgentSection").html(data);
	    			var tabVal = $("#filterCities").html();
	    			$("#otherAjaxCitiesOptionId").css("display", "inline-block");
	    			$('#agentDealingInTab').find('li').removeClass('active');
	    			$('#agentDealingInTab').append('<li id="otherCityTab" class="active"><a href="javascript://" style="cursor:auto;">'+tabVal+'</a></li>')
	    			//$("#ajaxLiId").html($("#filterCities").html());
	    		}
	    	}, async:true});
	    }
	}
	function clearFilterBuilder(){
		var flag = confirm('This will reset the data being displayed under Other Cities. Do you wish to proceed?');
		if(flag){
			$("#multiCityDD").hide();
			$("#normAgentSection").css("display", "inline-block");
			$(".normAgentSection").css("display", "inline-block");
			$("#otherCitiesOptionId").css("display", "inline-block");
			$("#otherCityTab").remove();
			$("#ajaxLiId").html("");
			$("#ajaxAgentSection").html("");
			$("#otherAjaxCitiesOptionId").css("display", "none");
			var otherCity = document.getElementsByName("otherCity");
			if(otherCity!=null){
				for (i = 0; i < otherCity.length; i++){
					otherCity[i].checked = false;
				}
			}
		}
		return flag;
	}

	function getOtherAgents(){
		var devId = '55237';
		var cities = [] ;
		$("#otherCity:checked").each(function() {
	        cities.push($(this).val());
	    });
	    if(cities.length == 0){
	    	alert('Please select at least one city');
	    }else{
	    	$("#multiCityDD").hide();
	    	ajaxService.getOtherAgents(devId, cities, displayingCity, {callback:function(data){
	    		if(data){
	    			$("#normAgentSection").css("display", "none");
	    			$(".normAgentSection").css("display", "none");
	    			$("#otherCitiesOptionId").css("display", "none");
	    			$("#ajaxAgentSection").html(data);
	    			var tabVal = $("#filterCities").html();
	    			$("#otherAjaxCitiesOptionId").css("display", "inline-block");
	    			$('#agentDealingInTab').find('li').removeClass('active');
	    			$('#agentDealingInTab').append('<li id="otherCityTab" class="active"><a href="javascript://" style="cursor:auto;">'+tabVal+'</a></li>')
	    			//$("#ajaxLiId").html($("#filterCities").html());
	    		}
	    	}, async:true});
	    }
	}
	function clearFilterBuilder(){
		var flag = confirm('This will reset the data being displayed under Other Cities. Do you wish to proceed?');
		if(flag){
			$("#multiCityDD").hide();
			$("#normAgentSection").css("display", "inline-block");
			$(".normAgentSection").css("display", "inline-block");
			$("#otherCitiesOptionId").css("display", "inline-block");
			$("#otherCityTab").remove();
			$("#ajaxLiId").html("");
			$("#ajaxAgentSection").html("");
			$("#otherAjaxCitiesOptionId").css("display", "none");
			var otherCity = document.getElementsByName("otherCity");
			if(otherCity!=null){
				for (i = 0; i < otherCity.length; i++){
					otherCity[i].checked = false;
				}
			}
		}
		return flag;
	}
