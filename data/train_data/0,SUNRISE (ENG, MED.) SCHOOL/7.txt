$brItemHideNumber = 0;

function brFooterResponsive(){
	$brFooterItems = $('.footer .pageContent > ul');
	$brFooterItem = $('.footer .pageContent > ul > li');
	$brFooterItemNumbers = $brFooterItem.length;
	$brFooterWidth = $('.footer .pageContent').width();
	
	if($(window).width() >= 1024)
	{
		$brFooterItem.removeAttr('style');
		$brFooterItems.removeAttr('style');
	}
	else if($(window).width() <= 1024 && $(window).width() > 800)
	{
		$brFooterItem.width($brFooterWidth/4);
		$brItemHideNumber = $brFooterItemNumbers - 4;
	}
	else if($(window).width() <= 800 && $(window).width() > 600)
	{		
		$brFooterItem.width($brFooterWidth/3);
		$brItemHideNumber = $brFooterItemNumbers - 3;
	}
	else if($(window).width() <= 600 && $(window).width() > 480)
	{		
		$brFooterItem.width($brFooterWidth/2);
		$brItemHideNumber = $brFooterItemNumbers - 2;
	}
	else if($(window).width() <= 480 && $(window).width() > 240)
	{
		$brFooterItem.width($brFooterWidth);		
		$brItemHideNumber = $brFooterItemNumbers - 1;
	}
	
	$brFooterItemWidth = $brFooterItem.width();
	$brFooterItems.width($brFooterItemNumbers*$brFooterItemWidth);
	$brFooterItemsWidth = $brFooterItems.width();
	
	if($(window).width() < $brFooterItemsWidth)
	{
		$('.brFooterSidebarCtr').addClass('show');
		$brLeft = 0;
	}
	else
	{
		$('.brFooterSidebarCtr').removeClass('show');
		$brFooterItems.removeAttr('style');
	}
}
brFooterResponsive();

$(window).resize(function(){
	$brFooterItems.stop().animate({ left: 0}, 1000, 'easeInOutQuint');
	brFooterResponsive();
});

$('.brFooterSidebarPrev').click(function(){
	if($(window).width() < $brFooterItemsWidth)
	{
		$brLeft -= $brFooterItemWidth;
		if($brLeft < $brItemHideNumber*$brFooterItemWidth*(-1))
		{
			$brLeft = 0;
		}
		$brFooterItems.stop().animate({ left: $brLeft}, 1000, 'easeInOutQuint');
	}
});

$('.brFooterSidebarNext').click(function(){
	if($(window).width() < $brFooterItemsWidth)
	{
		$brLeft += $brFooterItemWidth;	
		if($brLeft > 0)
		{
			$brLeft = -$brItemHideNumber*$brFooterItemWidth;				
		}
		$brFooterItems.stop().animate({ left: $brLeft}, 1000, 'easeInOutQuint');
	}
});

	{

		{
			$('#header').addClass('brSticky');
		}
		else

			$('#header').removeClass('brSticky');			
		}
	}
	else
	{
		if($(this).scrollTop() > 0)
		{
			$('#header').addClass('brSticky');
		}
		else
		{
			$('#header').removeClass('brSticky');			
		}
	}
}

});

function brNavigation(){
	$brNavPublicWidth = 0;	
	$('.navTabs .publicTabs .brNavTab').each(function(){
		$brNavPublicWidth += $(this).find('.navLink').outerWidth();
	});
	
	$brNavVisitorWidth = 0;	
	$('.navTabs .visitorTabs .brNavTab').each(function(){
		$brNavVisitorWidth += $(this).find('.navLink').outerWidth();
	});
	
	$brNavWidth = $('#navigation .pageContent').width();
	$brLogoBlockWidth = $('#logoBlock').outerWidth();
	$brSearchWidth = $('#searchBar').outerWidth();
	$brSpace = $brNavWidth - $brLogoBlockWidth - $brNavPublicWidth - $brNavVisitorWidth - $brSearchWidth - 5;
	
	if($brSpace < 30)
	{
		$('#navigation').addClass('BRJF_responsive');
	}
	else
	{
		$('#navigation').removeClass('BRJF_responsive');
		$('.publicTabs .brNavTab .tabLinks').removeAttr('style');
		$('html').animate({left: 0}, 300);			
	}
}

$brItemHideNumber = 0;

function brFooterResponsive(){
	$brFooterItems = $('.footer .pageContent > ul');
	$brFooterItem = $('.footer .pageContent > ul > li');
	$brFooterItemNumbers = $brFooterItem.length;
	$brFooterWidth = $('.footer .pageContent').width();
	
	if($(window).width() >= 1024)
	{
		$brFooterItem.removeAttr('style');
		$brFooterItems.removeAttr('style');
	}
	else if($(window).width() <= 1024 && $(window).width() > 800)
	{
		$brFooterItem.width($brFooterWidth/4);
		$brItemHideNumber = $brFooterItemNumbers - 4;
	}
	else if($(window).width() <= 800 && $(window).width() > 600)
	{		
		$brFooterItem.width($brFooterWidth/3);
		$brItemHideNumber = $brFooterItemNumbers - 3;
	}
	else if($(window).width() <= 600 && $(window).width() > 480)
	{		
		$brFooterItem.width($brFooterWidth/2);
		$brItemHideNumber = $brFooterItemNumbers - 2;
	}
	else if($(window).width() <= 480 && $(window).width() > 240)
	{
		$brFooterItem.width($brFooterWidth);		
		$brItemHideNumber = $brFooterItemNumbers - 1;
	}
	
	$brFooterItemWidth = $brFooterItem.width();
	$brFooterItems.width($brFooterItemNumbers*$brFooterItemWidth);
	$brFooterItemsWidth = $brFooterItems.width();
	
	if($(window).width() < $brFooterItemsWidth)
	{
		$('.brFooterSidebarCtr').addClass('show');
		$brLeft = 0;
	}
	else
	{
		$('.brFooterSidebarCtr').removeClass('show');
		$brFooterItems.removeAttr('style');
	}
}
brFooterResponsive();

$(window).resize(function(){
	$brFooterItems.stop().animate({ left: 0}, 1000, 'easeInOutQuint');
	brFooterResponsive();
});

$('.brFooterSidebarPrev').click(function(){
	if($(window).width() < $brFooterItemsWidth)
	{
		$brLeft -= $brFooterItemWidth;
		if($brLeft < $brItemHideNumber*$brFooterItemWidth*(-1))
		{
			$brLeft = 0;
		}
		$brFooterItems.stop().animate({ left: $brLeft}, 1000, 'easeInOutQuint');
	}
});

$('.brFooterSidebarNext').click(function(){
	if($(window).width() < $brFooterItemsWidth)
	{
		$brLeft += $brFooterItemWidth;	
		if($brLeft > 0)
		{
			$brLeft = -$brItemHideNumber*$brFooterItemWidth;				
		}
		$brFooterItems.stop().animate({ left: $brLeft}, 1000, 'easeInOutQuint');
	}
});
