        function shouldShowPoptop() {
          var sawrcpopup = parseInt(readCookie('sawrcpopup'));

          if(sawrcpopup) {
            writeCookie('sawrcpopup', sawrcpopup+1, 365);
            return (sawrcpopup+1) <= 3;
          }
          else {
            writeCookie('sawrcpopup', 1, 365);
            return true;
          }

          return false;
        }

        function writeCookie(cookieName, cookieValue, daysToExpire) {
            var expireTime = new Date();
            expireTime.setTime(expireTime.getTime() + (daysToExpire * 24 * 60 * 60 * 1000));
            document.cookie = cookieName + '=' + cookieValue + '; expires=' + expireTime.toUTCString() + '; path=/';
        }

        function readCookie(name) {
            var nameEQ = name + "=";
            var ca = document.cookie.split(';');
            for (var i = 0; i < ca.length; i++) {
                var c = ca[i];
                while (c.charAt(0) == ' ') c = c.substring(1, c.length);
                if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
            }
            return null;
        }
