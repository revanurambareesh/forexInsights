function initGMap() {
  var offLatLng = new google.maps.LatLng(38.0274, -78.4383);
  var map = new google.maps.Map(
    document.getElementById('map_canvas'), {
      center: offLatLng,
      zoom: 13,
      mapTypeId: google.maps.MapTypeId.ROADMAP
  });
  var marker = new google.maps.Marker({
      position: offLatLng,
      map: map
  });
  var jsAddr = "DISTRICT OFFICE<br> 1470 PANTOPS MTN PL<br>CHARLOTTESVILLE, VA 22911";
  var infowindow = new google.maps.InfoWindow({
    content: jsAddr
  });
  infowindow.open(map, marker);

  // Get the map's default panorama with defaults.
  panorama = map.getStreetView();
  panorama.setPosition(offLatLng);
  panorama.setVisible(false);       // false => no panorama
}

function toggleStreetView() {
  var toggle = panorama.getVisible();
  if (toggle == false) {
    panorama.setVisible(true);
  } else {
    panorama.setVisible(false);
  }
}
google.maps.event.addDomListener(window, 'resize', initGMap);
google.maps.event.addDomListener(window, 'load', initGMap);
