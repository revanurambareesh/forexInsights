var prevTag = "";
var currTag = "";
var prevAttr = "";
var currAttr = "";

$(document).ready(function(e) {

		$(".bluebox").hide();
		
        // Home Img Hover Effect
        $("#home-projects ul li a").on("mouseenter", function(event){
            event.preventDefault();
            var t = $(this).attr("class");
            $('#home-projects .backgrounds div').removeClass("active");
			$("#home-projects .backgrounds div." + t).addClass("active");
        }).stop().on("click", function(event){
			if( $(this).has('div') )
			{
				currTag = $(this);
				currAttr = $(this).attr('class');
				
				if(prevAttr == "")
				{
					currTag.children(".bluebox").slideDown('fast', function(){
							prevTag = currTag;
							prevAttr = currAttr;
							currAttr = "";
							currTag = "";
							return false;
						});
				}
				else if(prevAttr != currAttr)
				{
					prevTag.children(".bluebox").slideUp(4, function(){
							currTag.children(".bluebox").slideDown('fast', function(){
									prevTag = currTag;
									prevAttr = currAttr;
									currAttr = "";
									currTag = "";
									return false;
								});
						});
				}
				else
				{
					prevTag.children(".bluebox").slideUp('fast', function(){
							prevTag = "";
							prevAttr = "";
							currAttr = "";
							currTag = "";
							return false;
						});
				}
			}
		});

	/*==========================*/
	/*START : Touch detection code (supporting js - modernizr.touch.js) */	
    if (Modernizr.touch) {
		$('.dropdown-toggle').attr('data-toggle', 'dropdown');
	} else {
		$('.dropdown-toggle').attr('data-hover', 'dropdown');
	}
	/*ENDS : Touch detection code  */
	/*==========================*/
	
	/*-------------------------------------*/
	//STARTS : Resize div ( fit height width to screen )
	  resizeDiv();
	
	  window.onresize = function(event) {
		  resizeDiv();
	  }
	  function resizeDiv() {
		  //vpw = $(window).width(); 
		  vph = $(window).height();
		  $('.bg-blue').css({'height': vph + 'px'});
		  //$('.footer_copy4').css({'width': vpw + 'px'});
	  }
	//EndS : Resize div ( fit height width to screen )
	/*-------------------------------------*/

	/*-------------------------------------*/
	//STARTS : Footer tooltip
	$(".rbi").hover(function () {
		$(this).popover({
			content: "We hereby notify the public that the following Companies of Muthoot Pappachan Group are regulated by the Reserve Bank of India. <br /> 1. Muthoot Fincorp Ltd. <br /> 2. Muthoot Capital Services Ltd. <br />3. Muthoot Microfin Ltd.",
			placement : "top",
			html: true
		}).popover('show');
	}, function () {
		$(this).popover('hide');
	});
	//EndS : Footer tooltip
	/*-------------------------------------*/
	
	
	
});//END document ready
