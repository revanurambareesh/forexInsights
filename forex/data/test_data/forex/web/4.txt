        config.countrySelect.change( function() {
            // Get values from change event
            var $option = $( this ).children( 'option:selected' );
            var countryCode = $option.attr( 'id' );
            var countryName = $option.attr( 'value' );
            // Simply hide the restricted message each time to reset.
            $( '.restricted' ).hide();
            // If the countryCode is empty, then hide all links and set country banner to default.
            if( typeof countryCode == "undefined" ) {
                showCountryBanner( config.defaults.countryCode );
                return false;
            }
            // If it's one of the preselected countries at the top, slice 'pre_US' to just 'US'.
            if( countryCode.indexOf('pre_') != -1 ) {
                countryCode = countryCode.slice( 4 );
            }
            var countryDetails = getCountryDetails( countryCode );
            // Show the country banner associated with that country code.
            showCountryBanner( countryCode, countryDetails.countryCodeParent.toUpperCase() );
            // If the country selected is in the restricted countries list, then show restricted message
            if( countryDetails.restricted ){
                showRestrictedMessage( countryName );
            } else {
                $('.restricted').hide();
            }
            // Finally, pass the urls to the function to update the buttons.
            swapContent( countryDetails );
        });

        /**

        function showCountryBanner( countryCode, countryCodeParent ) {
            // Grab the country code if its not passed
            countryCode = typeof countryCode === "undefined" ? getUserCountry() : countryCode;
            var $banner;
            if( $('.banner[data-country="' + countryCode + '"]' )[0] ) {
                // Find a banner with that country code assigned.
                $banner = $('.banner[data-country="' + countryCode + '"]' );
            } else if( $('.banner[data-country="' + countryCodeParent + '"]' )[0] ) {
                $banner = $('.banner[data-country="' + countryCodeParent + '"]' );
            } else {
                // If no banner is found, set it to the default one.
                $banner = config.defaults.banner;
            }
            // Make sure the banner that is chosen isn't currently active. If not, hide it and fade in new one.
            if( !$banner.hasClass( 'active' ) ) {
                $( '.banner.active' ).hide().removeClass( 'active' );
                $banner.stop( true, true ).fadeIn().addClass( 'active' );
            }
        }
        /**
         * Swap out the home, live and demo urls for the links on the site
         */
        function swapContent( countryDetails ) {
            // Check if countryDetails are empty. If so, return.
            if( typeof countryDetails === "undefined" ) {
                return false;
            }
            // Set the demo and live links to disabled
            $( '.country-select-wrapper a.live, .country-select-wrapper a.demo' ).attr( 'href','javascript:void(0)' ).addClass( 'disabled' );

            if( countryDetails.demoURL ) {

            // Update currency Symbol

        /**
         * Get the user's country if it's passed by URL param or Geo Targeting
         */
        function getUserCountry() {
            var countryCode = "";
            // if country param is set in url, return that, otherwise check redirectData.
            if( fxcm.lib.getURLParameter( 'country' ) ) {
                countryCode = fxcm.lib.getURLParameter( 'country' );
            } else if( typeof redirectData === "object" && typeof redirectData.country === "string" && redirectData.country != "United States" ) {
                countryCode = redirectData.country;
            } else {
                countryCode = config.defaults.countryCode;
            }
            // Determine if the country code is either a value or ID in any of the available options.
            // If it is, return the id of that option
            if( config.countrySelect.find( "option[value='" + countryCode + "']" ) [0] ) {
                countryCode = config.countrySelect.find( "option[value='" + countryCode + "']:eq(0)" ).attr( 'id' );
            } else if( config.countrySelect.find( "option[id='" + countryCode.toUpperCase() + "']" ) [0] ) {
                countryCode = config.countrySelect.find( "option[id='" + countryCode.toUpperCase() + "']:eq(0)" ).attr( 'id' );
            } else {
                return config.defaults.countryCode;
            }
            // If the countrycode has a pre_ infront, just grab the ending code. ex: 'pre_US'
            if( countryCode.indexOf('pre_') != -1 ) {
                countryCode = countryCode.slice(4);
            }
            return countryCode;
        }
        /**

         */
        function getCountryDetails( countryCode ) {
            // Setup link variables
            var countryDetails = {
                'homeURL' : "",
                'demoURL' : "",
                'liveURL' : "",
                'countryCodeParent' : "",
                'currencySymbol' : config.defaults.currencySymbol,
                'restricted' : false

            // Loop through the urlMap to find the urls associated with that mapping.
            for( var map in urlMap ) {
                for( var lang in urlMap[map] ) {
                    var langMap = urlMap[map][lang];
                    var urls = urlMap[map]['urls'];
                    if( typeof langMap === 'object' && langMap.length ) {
                        // if countrycode is in the array of mappings, grab the urls.
                        if( $.inArray( countryCode, langMap ) > -1 ) {
                            // Set the urls
