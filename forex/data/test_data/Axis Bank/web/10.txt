	function validateForm() {
		if(document.loginForm.txtLoginId.value.length == 0) {
			alert('User Name should not be Blank!');
			document.loginForm.txtLoginId.select();
			document.loginForm.txtLoginId.focus();
			return false;
		}
		if(document.loginForm.txtLoginId.value.length < 4 || document.loginForm.txtLoginId.value.length >15) {
			alert('User Name should be 4..15 characters!');
			document.loginForm.txtLoginId.select();
			document.loginForm.txtLoginId.focus();
			return false;
		}
		if(document.loginForm.txtPassword.value.length == 0) {
			alert('Password should not be Blank!');
			document.loginForm.txtPassword.select();
			document.loginForm.txtPassword.focus();
			return false;
		}
		if(document.loginForm.txtPassword.value.length < 5 || document.loginForm.txtPassword.value.length >15) {
			alert('Password should be 5..15 characters!');
			document.loginForm.txtPassword.select();
			document.loginForm.txtPassword.focus();
			return false;
		}	
		return;
	}
