function appendToFunction(func, str)
{
    var nfs = "";
    if (func != null){
        var ofs = func.toString();
        var fcb = ofs.indexOf("{");
        var lcb = ofs.indexOf("}",fcb+1);
        var oss = ofs.substring(fcb + 1, lcb);
        //alert('h3=' + ofs + '||||\n' + oss);
        nfs = nbStripLeadingSemicolons(oss);
    }
	var s = nbStripLeadingSemicolons(nfs + ";" + str);
    var f = new Function(s);
    return f;
}
function nbStripLeadingSemicolons(s){
	while (s.substr(0,1) == ';' && s.length>1) {
    	s = s.substr(1);
    }
	return s;
}

function handleSubmitOnEnter(e)
{
    
    if (window.ghKeyPressHandler != null)
    {
        if ( typeof(ghKeyPressHandler == "function") )
        {
            ghKeyPressHandler(e);
        }
    }
    
    var keyPressed = '';

    var isMSIE = navigator.userAgent.indexOf('MSIE') != -1;

    if (isMSIE)
        keyPressed = String.fromCharCode(window.event.keyCode);
    else
        keyPressed = String.fromCharCode(e.which);

    if (keyPressed == "\r" || keyPressed == "\n")
    {
        
        var authenticationFormToUse = null;

        if (document.authenticationEntryFormForHeader && !document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryFormForHeader;
        }
        else if (!document.authenticationEntryFormForHeader && document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryForm;
        }

        if (authenticationFormToUse == null)
        {
            var headerFieldsFilledIn = 0;
            if (document.authenticationEntryFormForHeader)
            {
                if (typeof(document.authenticationEntryFormForHeader.screenName) != "undefined"
                    && document.authenticationEntryFormForHeader.screenName.value.length > 0)
                {
                    headerFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryFormForHeader.kclq) != "undefined"
                    && document.authenticationEntryFormForHeader.kclq.value.length > 0)
                {
                    headerFieldsFilledIn++;
                }
            }

            var bodyFieldsFilledIn = 0;
            if (document.authenticationEntryForm)
            {
                if (typeof(document.authenticationEntryForm.screenName) != "undefined"
                    && document.authenticationEntryForm.screenName.value.length > 0)
                {
                    bodyFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryForm.kclq) != "undefined"
                    && document.authenticationEntryForm.kclq.value.length > 0)
                {
                    bodyFieldsFilledIn++;
                }
            }

            if (headerFieldsFilledIn > bodyFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryFormForHeader;
            }

            if (bodyFieldsFilledIn > headerFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryForm;
            }
        }

        // If no form or a search is being performed
        if ( authenticationFormToUse == null || (typeof searchCtlr !== "undefined" && searchCtlr && document.forms["hcSearchForm"].token.value != '') )
        {
            window.document.onkeypress = doNothing;
        }
        else
        {
            if (authenticationFormToUse.kclq.value == "")
            {   
                authenticationFormToUse.kclq.focus();
            }
            else
            {
                authenticationFormToUse.submitButton.click();
    
                window.document.onkeypress = doNothing;
            }
        }


        return false;
    }

    return true;
}

    // Moves the comment input box to the position reply button that was clicked, rebinds needed events.
    // commentId = 0 refers to the top-most "reply to post" comment.
    // commentId's 1-n refer to "reply-to-other-comment" comments.
    function moveCommentInputBox(elmReply, commentId)
    {
        $(elmReply).removeAttr('onclick').unbind('click');

        // Get the id of the previous comment from the hidden form field.
        var oldCommentOn = $('#commentId').attr('value') || "";

        // Store the new id of what we are commenting on in the hidden form field.
        $('#commentId').attr('value', commentId);

        //calculate new size of textarea / profile image and reset the text
        var newWidth = $(elmReply).parent().width() - 70;

        var $commentInput = $('#commentInput');
        $commentInput.css('width', newWidth + "px");

        // EDL-9678
        // Don't show the "Leave a comment" text when the user clicks "reply". Instead, just set focus to the textarea.
        $commentInput.empty();
        $commentInput.removeAttr('onfocus').removeAttr('onblur');
        $commentInput.unbind('focus').unbind('blur');

        if (commentId == "0")
        {
            $('#commentInputDiv div.profileImage').removeClass('small').addClass('medium');
        }
        else
        {
            $('#commentInputDiv div.profileImage').removeClass('medium').addClass('small');
        }

        //swap the positions of the reply and comment input
        var elmCommentInputDiv  = $('#commentInputDiv');
        var elmPlaceHolder      = $('<div id=\"placeholder\" style=\"display:none;\"></div>');

        elmPlaceHolder.insertAfter($(elmReply));
        $(elmReply).remove()
                   .insertAfter(elmCommentInputDiv)
                   .bind('click', function() { moveCommentInputBox(elmReply, oldCommentOn); }); // Change the reply button's click event to make it move the comment box to where it was before.

        // Note: We use TWO focus() commands at the end of this chain because otherwise in IE 8 jQuery 1.3.2 will not always focus the textarea element properly.
        // For example, focus won't be set properly in IE 8 if the user clicks on a "reply" button above the current comment textarea.
        elmCommentInputDiv.remove()
                          .insertAfter(elmPlaceHolder)
                          .find('#commentInput').focus().focus();
        elmPlaceHolder.remove();

        return false;
    }

function handleSubmitOnEnter(e)
{
    
    if (window.ghKeyPressHandler != null)
    {
        if ( typeof(ghKeyPressHandler == "function") )
        {
            ghKeyPressHandler(e);
        }
    }
    
    var keyPressed = '';

    var isMSIE = navigator.userAgent.indexOf('MSIE') != -1;

    if (isMSIE)
        keyPressed = String.fromCharCode(window.event.keyCode);
    else
        keyPressed = String.fromCharCode(e.which);

    if (keyPressed == "\r" || keyPressed == "\n")
    {
        
        var authenticationFormToUse = null;

        if (document.authenticationEntryFormForHeader && !document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryFormForHeader;
        }
        else if (!document.authenticationEntryFormForHeader && document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryForm;
        }

        if (authenticationFormToUse == null)
        {
            var headerFieldsFilledIn = 0;
            if (document.authenticationEntryFormForHeader)
            {
                if (typeof(document.authenticationEntryFormForHeader.screenName) != "undefined"
                    && document.authenticationEntryFormForHeader.screenName.value.length > 0)
                {
                    headerFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryFormForHeader.kclq) != "undefined"
                    && document.authenticationEntryFormForHeader.kclq.value.length > 0)
                {
                    headerFieldsFilledIn++;
                }
            }

            var bodyFieldsFilledIn = 0;
            if (document.authenticationEntryForm)
            {
                if (typeof(document.authenticationEntryForm.screenName) != "undefined"
                    && document.authenticationEntryForm.screenName.value.length > 0)
                {
                    bodyFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryForm.kclq) != "undefined"
                    && document.authenticationEntryForm.kclq.value.length > 0)
                {
                    bodyFieldsFilledIn++;
                }
            }

            if (headerFieldsFilledIn > bodyFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryFormForHeader;
            }

            if (bodyFieldsFilledIn > headerFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryForm;
            }
        }

        // If no form or a search is being performed
        if ( authenticationFormToUse == null || (typeof searchCtlr !== "undefined" && searchCtlr && document.forms["hcSearchForm"].token.value != '') )
        {
            window.document.onkeypress = doNothing;
        }
        else
        {
            if (authenticationFormToUse.kclq.value == "")
            {   
                authenticationFormToUse.kclq.focus();
            }
            else
            {
                authenticationFormToUse.submitButton.click();
    
                window.document.onkeypress = doNothing;
            }
        }


        return false;
    }

    return true;
}

    // Moves the comment input box to the position reply button that was clicked, rebinds needed events.
    // commentId = 0 refers to the top-most "reply to post" comment.
    // commentId's 1-n refer to "reply-to-other-comment" comments.
    function moveCommentInputBox(elmReply, commentId)
    {
        $(elmReply).removeAttr('onclick').unbind('click');

        // Get the id of the previous comment from the hidden form field.
        var oldCommentOn = $('#commentId').attr('value') || "";

        // Store the new id of what we are commenting on in the hidden form field.
        $('#commentId').attr('value', commentId);

        //calculate new size of textarea / profile image and reset the text
        var newWidth = $(elmReply).parent().width() - 70;

        var $commentInput = $('#commentInput');
        $commentInput.css('width', newWidth + "px");

        // EDL-9678
        // Don't show the "Leave a comment" text when the user clicks "reply". Instead, just set focus to the textarea.
        $commentInput.empty();
        $commentInput.removeAttr('onfocus').removeAttr('onblur');
        $commentInput.unbind('focus').unbind('blur');

        if (commentId == "0")
        {
            $('#commentInputDiv div.profileImage').removeClass('small').addClass('medium');
        }
        else
        {
            $('#commentInputDiv div.profileImage').removeClass('medium').addClass('small');
        }

        //swap the positions of the reply and comment input
        var elmCommentInputDiv  = $('#commentInputDiv');
        var elmPlaceHolder      = $('<div id=\"placeholder\" style=\"display:none;\"></div>');

        elmPlaceHolder.insertAfter($(elmReply));
        $(elmReply).remove()
                   .insertAfter(elmCommentInputDiv)
                   .bind('click', function() { moveCommentInputBox(elmReply, oldCommentOn); }); // Change the reply button's click event to make it move the comment box to where it was before.

        // Note: We use TWO focus() commands at the end of this chain because otherwise in IE 8 jQuery 1.3.2 will not always focus the textarea element properly.
        // For example, focus won't be set properly in IE 8 if the user clicks on a "reply" button above the current comment textarea.
        elmCommentInputDiv.remove()
                          .insertAfter(elmPlaceHolder)
                          .find('#commentInput').focus().focus();
        elmPlaceHolder.remove();

        return false;
    }
