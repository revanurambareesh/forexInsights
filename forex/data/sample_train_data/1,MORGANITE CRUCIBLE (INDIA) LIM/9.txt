        function changeDataGroupping(e) {

            for (i = 0; i < chart.series.length; i++) {

                    dataGroupEnabled = false;
                }
                else if (e.rangeSelectorButton.text == '3m') {
                    dataGroupEnabled = true;
                    dataGroupUnits = [['day', [1]]];
                }
                else if (e.rangeSelectorButton.text == 'YTD' && daysDiff <= 30) {
                    dataGroupEnabled = false;
                }
                else if (e.rangeSelectorButton.text == 'YTD') {
                    dataGroupEnabled = true;
                    dataGroupUnits = [['day', [1]]];
                }
                else if (e.rangeSelectorButton.text == '1y') {
                    dataGroupEnabled = true;
                    dataGroupUnits = [['day', [1]]];
                }
                else if (e.rangeSelectorButton.text == 'All' && daysDiff <= 30) {
                    dataGroupEnabled = false;
                }
                else if (e.rangeSelectorButton.text == 'All' && daysDiff <= 730) {
                    dataGroupEnabled = true;
                    dataGroupUnits = [['day', [1]]];
                }
                else {
                    dataGroupEnabled = true;
                    dataGroupUnits = [['week', [1]]];
                }

                if (dataGroupEnabled == true) {
                    $("#container").highcharts().series[i].update({
                        dataGrouping: {
                            enabled: dataGroupEnabled,
                            units: dataGroupUnits
                        }
                    });
                }
                else {
                    $("#container").highcharts().series[i].update({
                        dataGrouping: {
                            enabled: dataGroupEnabled
                        }
                    });
                }
            }

            $('#container').highcharts().hideLoading();
        }

        function stringToColour(str) {
            var colour = '#';
            colorsUsed = colorsUsed + 1;
            if (colorsUsed <= 9) {
                colour = getColor[colorsUsed];
            }
            else {
                var hash = 0;
                for (var i = 0; i < str.length; i++) {
                    hash = str.charCodeAt(i) + ((hash << 5) - hash);
                }
                for (var i = 0; i < 3; i++) {
                    var value = (hash >> (i * 8)) & 0xFF;
                    colour += ('00' + value.toString(16)).substr(-2);
                }
            }
            return colour;
        }

        function stringToColour(str) {
            var colour = '#';
            colorsUsed = colorsUsed + 1;
            if (colorsUsed <= 9) {
                colour = getColor[colorsUsed];
            }
            else {
                var hash = 0;
                for (var i = 0; i < str.length; i++) {
                    hash = str.charCodeAt(i) + ((hash << 5) - hash);
                }
                for (var i = 0; i < 3; i++) {
                    var value = (hash >> (i * 8)) & 0xFF;
                    colour += ('00' + value.toString(16)).substr(-2);
                }
            }
            return colour;
        }
