	function formatParams(str){
		new_str = str.replace(/ /g,"-");
		new_str = new_str.replace(/&/g,'and');
		return new_str.toLowerCase();
	}	

	function buildURL(){
		var params 		= '';
		var dest 		= $('select#srch_destn').val();
		var type 		= $('select#srch_type').val();
		var dep_month 	= $('#srch_dep_month').val();
		var price_range = $('select#srch_price_range').val();

		/*----- Start: building the URL with parameters ------*/
		if(dest.length>0){
			params+='/destination/'+formatParams(dest);
			}
			if(type.length>0){
				params+='/tourType/'+formatParams(type);
			}
			if(dep_month.length>0){
				params+='/departureMonth/'+ dep_month.replace(/-/g," ");
			}
			if(price_range.length>0){
				params+='/priceRange/'+formatParams(price_range);
			}
				//params+='/comefor/'+formatParams(price_range);
		/*----- End: building the URL with parameters ------*/
		return params;
	}

	function formatParams(str){
		new_str = str.replace(/ /g,"-");
		new_str = new_str.replace(/&/g,'and');
		return new_str.toLowerCase();
	}	

	function buildURL(){
		var params 		= '';
		var dest 		= $('select#srch_destn').val();
		var type 		= $('select#srch_type').val();
		var dep_month 	= $('#srch_dep_month').val();
		var price_range = $('select#srch_price_range').val();

		/*----- Start: building the URL with parameters ------*/
		if(dest.length>0){
			params+='/destination/'+formatParams(dest);
			}
			if(type.length>0){
				params+='/tourType/'+formatParams(type);
			}
			if(dep_month.length>0){
				params+='/departureMonth/'+ dep_month.replace(/-/g," ");
			}
			if(price_range.length>0){
				params+='/priceRange/'+formatParams(price_range);
			}
				//params+='/comefor/'+formatParams(price_range);
		/*----- End: building the URL with parameters ------*/
		return params;
	}
