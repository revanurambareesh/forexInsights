                                    $(this).addClass('rowHead');
                                }
                            });

                            $('<div class="mdcClear"></div>').insertAfter(".rowHead");

                        SetTops: function (ctrls) {
                            var topArray = this.GetTopMaxArray(ctrls);
                            for (var i = 0; i < ctrls.length; i++) {
                                var ctrl = ctrls.eq(i);
                                if (ctrl.find('.line').length != 0) {
                                    var arrayIndex = parseInt(i / 3);
                                    if (topArray[arrayIndex] > ctrl.find('.line').position().top) {
                                        var height = topArray[arrayIndex] + 8;
                                        ctrl.find('.line').offset({ top: height });
                                    }
                                }
                            }
                        },
                        GetTopMaxArray: function (ctrls) {
                            var topArrays = [];

                                if (topArrays[parseInt(i / 3)] == undefined) {
                                    topArrays[parseInt(i / 3)] = 0;
                                }

                                if (ctrl.find('.line').length != 0) {
                                    if (ctrl.find('.line').position().top > topArrays[parseInt(i / 3)]) {
                                        topArrays[parseInt(i / 3)] = ctrl.find('.line').position().top;
                                    }
                                }
                            }
                            return topArrays;
