        var emailid = "cliser@hdfcfund.com"
        lblEnquiryMsg.innerHTML = "";

        txtEmailId.className = txtEmailId.className.replace(/(?:^|\s)HighlightBkgRed(?!\S)/g, '');
        txtMobileNo.className = txtMobileNo.className.replace(/(?:^|\s)HighlightBkgRed(?!\S)/g, '');
      
        //        if (!validchars(txtName) ||
        //       !validchars(txtEmailId)
        //           ) {
        //            return false;

        //        if (!CheckBlank(txtName, 'Name ')
        //             || !CheckAlphaSpecialChar(txtName, "Name", " '")
        //             || !CheckLSpace(txtName, 'Name ')
        //             ) {
        //            return false;
        //        }
        //        if (!CheckBlank(txtEmailId, 'Email ID ')
        //             || !CheckMail(txtEmailId, 'Email ID ')
        //             ) {
        //            return false;
        //        }
        //        if ((txtEmailId.value).substring(lat - 1, lat) == dot || (txtEmailId.value).substring(lat + 1, lat + 2) == dot) {

        //            return false;
        //        }


        //        if (!CheckBlank(txtMobileNo, "Mobile No")
        //                   || !CheckLSpace(txtMobileNo, "Mobile No. ")
        //                   || !CheckLSpace(txtMobileNo, 'Mobile No')
        //                   || !CheckMobile(txtMobileNo, 'Mobile No')
        //                   || !MobileCheckLength_New(txtMobileNo, 'Mobile No')) {
        //            return false;
        //        }

        //        if (!CheckDropDown(ddlState, 'State ')
        //		      || !CheckDropDown(ddlCity, 'City ')
        //		      ) {
        //            return false;
        //        }

        if (!CheckValidChars(txtName)) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckValidChars(txtEmailId)) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckForBlank(txtName, 'Name ')) {
            lblEnquiryMsg.innerHTML = "Please enter your Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForAlphaSpecialChar(txtName, "Name", " '")) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForLSpace(txtName, 'Name ')) {
            lblEnquiryMsg.innerHTML = "Name " + " : Spaces at the start are not allowed";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForBlank(txtEmailId, 'Email ID ')) {
            lblEnquiryMsg.innerHTML = "Please enter your Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForMail(lblEnquiryMsg, txtEmailId, 'Email ID ')) {
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }

        if ((txtEmailId.value).substring(lat - 1, lat) == dot || (txtEmailId.value).substring(lat + 1, lat + 2) == dot || (txtEmailId.value == emailid)) {
            //            lblEnquiryMsg.innerHTML = "Please enter a valid Email Address";
            lblEnquiryMsg.innerHTML = "Please enter your Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }


        if (!CheckForBlank(txtMobileNo, "Mobile No")) {
            lblEnquiryMsg.innerHTML = "Please enter your Mobile Number";
            txtMobileNo.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForLSpace(txtMobileNo, 'Mobile No')) {
            lblEnquiryMsg.innerHTML = "Mobile No " + " : Spaces at the start are not allowed";
            txtMobileNo.className += " HighlightBkgRed";
            // alert(strControlName + " : Spaces at the start are not allowed");
            return false;
        }
        if (!CheckForMobile(txtMobileNo, 'Mobile No')) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Mobile Number";
            txtMobileNo.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckMobileLength(txtMobileNo, 'Mobile No')) {
            //alert(strControlName+ ":It should 10 digits number");
            lblEnquiryMsg.innerHTML = "Mobile No " + ": It should be 10 digits number";

            return false;
        }

            return false;

        if (!CheckForDropDown(ddlCity, 'City ')) {
            lblEnquiryMsg.innerHTML = "Please select an Indian City from the list.";
            return false;
        }
        return true;

    function ValidSearchTextBox() {
        var txtSearch = document.getElementsByClassName('clstxtsrch')[0];
        if (txtSearch.value == "Search")
            txtSearch.value = "";
        if (!CheckBlank(txtSearch, "Search Keyword") ||
            !CheckSpecialChar(txtSearch, "Search Keyword", " ,'()-") ||
            !charlencheck(txtSearch, "Search Keyword", 3)) {
            return false;
        }
        else {
            return true;
        }
    }


    function AddToFavourites(Current, PlanID, SchemeID, SchemeName, PlanType, Date, NAVAmount) {
        if (PlanID == "" || PlanID == null) {
            alert("This plan cannot be added to favourites");
            return false;
        }
        SchemeName = SchemeName.replace("$", "'");

        Current = $('.bottomDv').find("a.favIcon[value='" + PlanID + "']");
        if ($('.favouritesList').find('li').length == 8 && !$(Current).hasClass("actFav")) {
            //alert("Maximum 8 plans can only be added.Please remove other plans to add");
            $('.overLay, .lightBoxHold, .favouriteBox').fadeIn(200);
        }
        else if (!$(Current).hasClass("actFav") && $(".favouritesList ul li[value='" + PlanID + "']").text().length > 0) {
            // alert("This plan has already been to Favourites");

        }
        else {
            if ($(Current).hasClass("actFav")) {
                //remove active and li
                $(Current).removeClass("actFav");

                $(".favouritesList ul li[value='" + PlanID + "']").remove();
            }
            else {
                // $('.inrTbl').find("a.fav[value='" + PlanID + "']").addClass("actFav");
                $(Current).addClass("actFav");
                var newLi = "<li value=" + PlanID + ">" +
        "<div class='fundName'>" +
        "<h3> " + SchemeName + "</h3>" +
        "<div class='closeBtnFavourites'" + "value=" + PlanID + ">" +
        "<img src='../../Includes/images1/closeButton.gif' alt='' /></div>" +
        "</div>" +
        "<div class='fundDtls'>" +
        "<h4>" + PlanType + "</h4>" +
        "<p>" +
        "NAV Date : " + Date + "</p>" +
        "<p>" +
        "Nav : " + NAVAmount + "</p>" +
        "</div>" +
        "</li>";
                $(".favouritesList ul").append(newLi);
            }

            $(".FavCount").text($('.favouritesList').find('li').length);
            UpdateFavouritesCookie();
        }
        return false;
    }

        if (!CheckValidChars(txtName)) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckValidChars(txtEmailId)) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckForBlank(txtName, 'Name ')) {
            lblEnquiryMsg.innerHTML = "Please enter your Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForAlphaSpecialChar(txtName, "Name", " '")) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Name";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForLSpace(txtName, 'Name ')) {
            lblEnquiryMsg.innerHTML = "Name " + " : Spaces at the start are not allowed";
            txtName.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForBlank(txtEmailId, 'Email ID ')) {
            lblEnquiryMsg.innerHTML = "Please enter your Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForMail(lblEnquiryMsg, txtEmailId, 'Email ID ')) {
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }

        if ((txtEmailId.value).substring(lat - 1, lat) == dot || (txtEmailId.value).substring(lat + 1, lat + 2) == dot || (txtEmailId.value == emailid)) {
            //            lblEnquiryMsg.innerHTML = "Please enter a valid Email Address";
            lblEnquiryMsg.innerHTML = "Please enter your Email Address";
            txtEmailId.className += " HighlightBkgRed";
            return false;
        }


        if (!CheckForBlank(txtMobileNo, "Mobile No")) {
            lblEnquiryMsg.innerHTML = "Please enter your Mobile Number";
            txtMobileNo.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForLSpace(txtMobileNo, 'Mobile No')) {
            lblEnquiryMsg.innerHTML = "Mobile No " + " : Spaces at the start are not allowed";
            txtMobileNo.className += " HighlightBkgRed";
            // alert(strControlName + " : Spaces at the start are not allowed");
            return false;
        }
        if (!CheckForMobile(txtMobileNo, 'Mobile No')) {
            lblEnquiryMsg.innerHTML = "Please enter a valid Mobile Number";
            txtMobileNo.className += " HighlightBkgRed";
            return false;
        }
        if (!CheckMobileLength(txtMobileNo, 'Mobile No')) {
            //alert(strControlName+ ":It should 10 digits number");
            lblEnquiryMsg.innerHTML = "Mobile No " + ": It should be 10 digits number";
            txtMobileNo.className += " HighlightBkgRed";
            return false;
        }

        if (!CheckForDropDown(ddlState, 'State ')) {
            lblEnquiryMsg.innerHTML = "Please select an Indian State from the list.";
            return false;
        }
        if (!CheckForDropDown(ddlCity, 'City ')) {
            lblEnquiryMsg.innerHTML = "Please select an Indian City from the list.";
            return false;
        }
        return true;
    }

    function OpenLinks() {

        if (urlVal != "") {
