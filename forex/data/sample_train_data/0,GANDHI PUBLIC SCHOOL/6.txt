             function checkIsMobile(){
    var browser={    
        versions:function(){          
            var u = navigator.userAgent, app = navigator.appVersion;            
            return {                               
                mobile : !!u.match(/AppleWebKit.*Mobile/) || !!u.match(/Windows Phone/) || !!u.match(/Android/) || !!u.match(/MQQBrowser/),
                iPad: u.indexOf('iPad') > -1
            };
        }()
    } 
    if(browser.versions.mobile && !browser.versions.iPad && window.screen.width<767){
        return true;
    }else{
        return false;
    }
}

function popup_video(e){
    e = e || window.event;
    e = $.event.fix(e);

    e.preventDefault();
    e.stopPropagation();

    var image;
    if ($(e.target).find('img').length) {
        image = $(e.target).find('img')[0];
    } else {
        image = $(e.target).parent().parent().find('img')[0];
    }

    $('#frame').attr('src', '//www.youtube.com/embed/'+image.src.split('/')[4]);
    $('#youtube_title').text(image.alt);
    $('#youtube_description').text(image.title);

    $('.video_dialog').show(0);

    var resizeInterval = setInterval(function(){
        if ($('.video_dialog_container').width() !== $(window).width()) {
            popup_resize();
        }
        if ($(window).width() > $(window).height()) {
            if ($('#youtube_title').height()+$('#frame').height()+ 20 !== $(window).height()) {
                popup_resize();
            }
        }
    }, 300);

    $('.close').click(function(){
        $('.video_dialog').hide(0);
        $('#frame').attr('src', '');
        clearInterval(resizeInterval);
    });
}

             function checkIsMobile(){
    var browser={    
        versions:function(){          
            var u = navigator.userAgent, app = navigator.appVersion;            
            return {                               
                mobile : !!u.match(/AppleWebKit.*Mobile/) || !!u.match(/Windows Phone/) || !!u.match(/Android/) || !!u.match(/MQQBrowser/),
                iPad: u.indexOf('iPad') > -1
            };
        }()
    } 
    if(browser.versions.mobile && !browser.versions.iPad && window.screen.width<767){
        return true;
    }else{
        return false;
    }
}

function popup_video(e){
    e = e || window.event;
    e = $.event.fix(e);

    e.preventDefault();
    e.stopPropagation();

    var image;
    if ($(e.target).find('img').length) {
        image = $(e.target).find('img')[0];
    } else {
        image = $(e.target).parent().parent().find('img')[0];
    }

    $('#frame').attr('src', '//www.youtube.com/embed/'+image.src.split('/')[4]);
    $('#youtube_title').text(image.alt);
    $('#youtube_description').text(image.title);

    $('.video_dialog').show(0);

    var resizeInterval = setInterval(function(){
        if ($('.video_dialog_container').width() !== $(window).width()) {
            popup_resize();
        }
        if ($(window).width() > $(window).height()) {
            if ($('#youtube_title').height()+$('#frame').height()+ 20 !== $(window).height()) {
                popup_resize();
            }
        }
    }, 300);

    $('.close').click(function(){
        $('.video_dialog').hide(0);
        $('#frame').attr('src', '');
        clearInterval(resizeInterval);
    });
}
