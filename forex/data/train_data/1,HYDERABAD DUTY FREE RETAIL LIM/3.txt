        function validCaptcha() {

            var response = grecaptcha.getResponse();

            if (response.length == 0) {

                return false;
            }
            else {
                return true;
            }

        function validate() {

            var validCap = validCaptcha();

            if (validCap) {


                var name, mobile, email, select, dropdown, textbox;

                name = document.forms["form1"]["txtname"].value;
                mobile = document.forms["form1"]["txtmobile"].value;
                email = document.forms["form1"]["txtemail"].value;
                strmydropdown = document.forms["form1"]["mydropdown"].value;
                textbox = document.forms["form1"]["ftextbox"].value;

                if (name == "") {
                    alert("Please enter your name");
                    document.getElementById("txtname").focus();
                    return false;
                }

                if (mobile == "") {
                    alert("Enter your phone number");
                    document.getElementById("txtmobile").focus();
                    return false;
                }
                if (!mobile.toString().match(/^[-]?\d*\.?\d*$/)) {
                    alert("Please enter valid phone number");
                    document.getElementById("txtmobile").focus();
                    return false;
                }
                if (mobile.length != 10) {
                    alert("Please enter 10 digit phone number.");
                    document.getElementById("txtmobile").focus();
                    return false;
                }

                if (email == "") {
                    alert("Please provide your Email!");
                    document.getElementById("txtemail").focus();
                    return false;
                }
                if (!validateEmail(email)) {
                    alert("Please enter valid Email Id");
                    document.getElementById("txtemail").focus();
                    return false;
                }

                else {
                    // DO NOTHING
                }
                if (email == "") {
                    alert("Please provide your Email!");
                    document.getElementById("txtemail").focus();
                    return false;
                }


                if (strmydropdown.length == 0 || strmydropdown == " " || strmydropdown == "Select Type") {
                    alert("Please select anyone");
                    return false;
                }

                if (textbox == "") {
                    alert("Please enter your details");
                    document.getElementById("ftextbox").focus();
                    return false;
                }

            }

            else {

                alert("Please check the captcha");
                $(".g-recaptcha").focus();
                return false;

            }

        }

            return re.test(email);
        }
