$(function() {
	var $menu = $('#district-men);
	var $list = $('#district-menu .school-container');
	var $schools = $list.find('.school');
	var $tabs = $('#district-menu .school-type-tabs li');
	var $search = $('#district-menu .school-search');
	var $searchbutton = $search.find('.search-button');
	var $searchinput = $search.find('.search-input input');
	
	$menu.click(function(event){
		event.stopPropagation();
	});
	
	$tabs.click(function(event) {
		event.preventDefault();
		var $this=$(this);
		if($this.hasClass('active')) {
			$this.removeClass('active');
			$schools.addClass('active');
		} else {
			$tabs.removeClass('active');
			$this.addClass('active');
			$schools.removeClass('active');
			$schools.filter($this.attr('data-filter')).addClass('active');
			//console.log($this.attr('data-filter'), $tabs.filter($this.attr('data-filter')));
		}
	});
	
	var cancelSearch = function() {
		$searchinput.val('');
		$schools.addClass('active');
		$search.removeClass('open');
	}
	
	$searchinput.keyup(function(event){
		if(event.keyCode == 27) { // escape
			event.stopPropagation();
			cancelSearch();
			return;
		}
		var valThis = $(this).val().toLowerCase();
		if(valThis == ""){
			$schools.removeClass('active');
		} else {
			$schools.each(function(){
				var $this = $(this);
				var text = $this.find('.title').text().toLowerCase();
				$this.toggleClass('active', (text.indexOf(valThis) >= 0));
			});
		};
	});
	
	$searchbutton.click(function(event) {
		event.preventDefault();
		
		if($search.hasClass('open')) {
			cancelSearch();
		} else {
			$search.addClass('open');
			$searchinput.focus();
			$schools.removeClass('active');
		}
	});
});

$(function() {
	var $menu = $('#district-men);
	var $list = $('#district-menu .school-container');
	var $schools = $list.find('.school');
	var $tabs = $('#district-menu .school-type-tabs li');
	var $search = $('#district-menu .school-search');
	var $searchbutton = $search.find('.search-button');
	var $searchinput = $search.find('.search-input input');
	
	$menu.click(function(event){
		event.stopPropagation();
	});
	
	$tabs.click(function(event) {
		event.preventDefault();
		var $this=$(this);
		if($this.hasClass('active')) {
			$this.removeClass('active');
			$schools.addClass('active');
		} else {
			$tabs.removeClass('active');
			$this.addClass('active');
			$schools.removeClass('active');
			$schools.filter($this.attr('data-filter')).addClass('active');
			//console.log($this.attr('data-filter'), $tabs.filter($this.attr('data-filter')));
		}
	});
	
	var cancelSearch = function() {
		$searchinput.val('');
		$schools.addClass('active');
		$search.removeClass('open');
	}
	
	$searchinput.keyup(function(event){
		if(event.keyCode == 27) { // escape
			event.stopPropagation();
			cancelSearch();
			return;
		}
		var valThis = $(this).val().toLowerCase();
		if(valThis == ""){
			$schools.removeClass('active');
		} else {
			$schools.each(function(){
				var $this = $(this);
				var text = $this.find('.title').text().toLowerCase();
				$this.toggleClass('active', (text.indexOf(valThis) >= 0));
			});
		};
	});
	
	$searchbutton.click(function(event) {
		event.preventDefault();
		
		if($search.hasClass('open')) {
			cancelSearch();
		} else {
			$search.addClass('open');
			$searchinput.focus();
			$schools.removeClass('active');
		}
	});
});
