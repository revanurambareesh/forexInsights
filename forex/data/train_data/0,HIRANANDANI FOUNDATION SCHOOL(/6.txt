    function validation() {
        var emailRegex = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
        var $emailreg = emailRegex.test($email.val());
        var $mobilereg = /^[0-9-+]+$/;
        var $mobileval = $mobilereg.test($mobile.val());



        if ($name.val() == "" || $name.val() == "Enter Name") {
            $('.comonerror').html("Please enter your name").css("display", "inline-block");
            return false;
        }

        if ($email.val() == "" || $email.val() == "Enter Email") {
            $('.comonerror').html("Please enter your email id").css("display", "inline-block");
            return false;
        }
        if ($emailreg == false) {
            $('.comonerror').html("Please enter your correct email id").css("display", "inline-block");
            return false;
        }
        if ($mobile.val() == "" || $mobile.val() == "Enter Mobile") {
            $('.comonerror').html("Please enter your mobile no.").css("display", "inline-block");
            return false;
        }
        if ($mobile.val().length < 6 || $mobile.val().length > 13) {
            $('.comonerror').html("Please enter your correct mobile no.").css("display", "inline-block");
            return false;
        }

        if ($txtAddress.val() == "") {
            $('.comonerror').html("Please enter your address").css("display", "inline-block");
            return false;
        }

        if ($mobileval == false) {
            $('.comonerror').html("Only digits allowed in mobile no.").css("display", "inline-block");
            return false;
        }

        if ($city.val() == "") {
            $('.comonerror').html("Please enter your city").css("display", "inline-block");
            return false;
        }

        if ($country.val() == "") {
            $('.comonerror').html("Please enter your country").css("display", "inline-block");
            return false;
        }
      

        if ($particularproj.val() == "---Select Project---") {
            $('.comonerror').html("Please select at least one project").css("display", "inline-block");
            return false;
        }

        if ($captcha.val() == "") {
            $('.comonerror').html("Please enter verification code").css("display", "inline-block");
            return false;
        }
        else {
            //$(':text').removeClass('watermark').addClass('whiteborder');
            $('.comonerror').hide();
            //SendEnquiry();
            jj(commonProject);
            return true;
        }

    }

    function validation() {
        var emailRegex = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
        var $emailreg = emailRegex.test($email.val());
        var $mobilereg = /^[0-9-+]+$/;
        var $mobileval = $mobilereg.test($mobile.val());



        if ($name.val() == "" || $name.val() == "Enter Name") {
            $('.comonerror').html("Please enter your name").css("display", "inline-block");
            return false;
        }

        if ($email.val() == "" || $email.val() == "Enter Email") {
            $('.comonerror').html("Please enter your email id").css("display", "inline-block");
            return false;
        }
        if ($emailreg == false) {
            $('.comonerror').html("Please enter your correct email id").css("display", "inline-block");
            return false;
        }
        if ($mobile.val() == "" || $mobile.val() == "Enter Mobile") {
            $('.comonerror').html("Please enter your mobile no.").css("display", "inline-block");
            return false;
        }
        if ($mobile.val().length < 6 || $mobile.val().length > 13) {
            $('.comonerror').html("Please enter your correct mobile no.").css("display", "inline-block");
            return false;
        }

        if ($txtAddress.val() == "") {
            $('.comonerror').html("Please enter your address").css("display", "inline-block");
            return false;
        }

        if ($mobileval == false) {
            $('.comonerror').html("Only digits allowed in mobile no.").css("display", "inline-block");
            return false;
        }

        if ($city.val() == "") {
            $('.comonerror').html("Please enter your city").css("display", "inline-block");
            return false;
        }

        if ($country.val() == "") {
            $('.comonerror').html("Please enter your country").css("display", "inline-block");
            return false;
        }
      

        if ($particularproj.val() == "---Select Project---") {
            $('.comonerror').html("Please select at least one project").css("display", "inline-block");
            return false;
        }

        if ($captcha.val() == "") {
            $('.comonerror').html("Please enter verification code").css("display", "inline-block");
            return false;
        }
        else {
            //$(':text').removeClass('watermark').addClass('whiteborder');
            $('.comonerror').hide();
            //SendEnquiry();
            jj(commonProject);
            return true;
        }

    }

        function fnOpenPrint(data) {
            var openPrint = window.open(data);
            openPrint.onload = function () {
                var doc = openPrint.document;
                var newDiv = doc.createElement("button");
                newDiv.innerHTML = "Print";
                newDiv.setAttribute("style", "background-color: #4286F4;border: medium none;color: white; cursor: pointer;font-family: lato;font-size: 14px;padding: 10px;position: absolute;right: 10px;top: 10px;z-index: 2000;");

                doc.body.appendChild(newDiv);
                newDiv.onclick = function () {
                    openPrint.print();
                };
            };
        }
