                        function valiadte_home_callback() {
                            var validator = $("#home_callback").validate({
                                rules: {contact: {required: true, number: true, minlength: 10, maxlength: 12}},
                                message: {
                                    contact: {required: "Enter your contact number",
                                        number: "only number is required",
                                        minlength: "Enter atleast 10 digits",
                                        maxlength: "More than 12 digits not allowed"
                                    }
                                },
                                submitHandler: function (form) {
                                    var number = $("#contact").val();
                                    var enq_campaign = $("#home_callback_campaign").val();
                                    var csrf_test_name = $("input[name=token_key]").val();
                                    var name = '';
                                    $.ajax({
                                        type: 'post',
                                        dataType: 'json',
                                        url: window.location.origin + "/home/getCallback",
                                        data: {'number': number, 'req_campaign': enq_campaign, 'call_back_name': name, 'token_key': csrf_test_name},
                                        success: function (data) {
                                            var status = data.status;
                                            if (status == 1) {
                                                common_coversion_code();
                                                setTimeout(function () {
                                                    execute_conversion_code(enq_campaign);
                                                }, 3000);
                                                $("#callback_msg").html(data.msg);
                                                $("#contact").val('');
                                                setTimeout(function () {
                                                    $('#callback_msg').html('');
                                                }, 4000);
                                            } else if (status == 0) {
                                                $("#callback_msg").html(data.msg);
                                                return false;
                                            }
                                        }
                                    });
                                }
                            });
                        }
