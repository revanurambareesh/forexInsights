		for(var c = 0; c < arrCbox.length; c++) {
			if(arrCbox[c] == 'ARTICLE')

			else if(arrCbox[c] == 'ISI')
          		delete recPub[i].b;
          	else if(arrCbox[c] == 'SCOPUS')
          		delete recPub[i].c;

          		delete recPub[i].d;
			else if(arrCbox[c] == 'CPBOOK')
          		delete recPub[i].e;	
    	}  
    }  
    
  	return recPub;
};
