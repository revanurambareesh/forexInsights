$(document).ready(function() {
	function aniamte_logos_retail(){
		var i=1;
		$delay = 0;
		var k=0;
		$('.retail-animation').each(function() {
		  	var $this = $(this);
			if(k==1){$delay= 600;}else if(k==2){$delay= 1500;}else if(k==3){$delay= 2100;}
			k++;
		  setTimeout(function() {
		  //$this.addClass('animate');
			if(i==1){$this.addClass('sales-today-text-animation');}
			else if(i==2){$this.addClass('sales-today-amount-animation')}
			else if(i==3){$this.addClass('grossmargin-text-animation')}
			else if(i==4){$this.addClass('grossmargin-amount-animation')}
			i++;
		  }, $delay);
		});
	  }
	  function aniamte_restaurant_smile(){
		var i=1;
		$delay2 = 0;
		  var j=0;
		$('.restaurant-smile').each(function() {
		  var $this = $(this);
			j++;
		  $delay2 += 200;if(j==6){$delay2 += 150;}
		  setTimeout(function() {
			//$this.addClass('animate');
			if(i==1){$this.addClass('restaurant-smile1-animation');}
			else if(i==2){$this.addClass('restaurant-smile2-animation');}
			else if(i==3){$this.addClass('restaurant-smile3-animation');}
			else if(i==4){$this.addClass('restaurant-smile4-animation');}
			else if(i==5){$this.addClass('restaurant-smile5-animation');}
			else if(i==6){$this.addClass('home-my-pulse-img-animation');}
			i++;
		  }, $delay2);
		});
	  }
	  function aniamte_dist_logo(){
		var i=1;
		$delay3 = 0;
		$('.dist-animation').each(function() {
		  var $this = $(this);
		  $delay3 += 250;
		  setTimeout(function() {
			//$this.addClass('animate');
			if(i==1){$this.addClass('dist-animation-logo1-animate');}
			else if(i==5){$this.addClass('dist-animation-logo4-animate');}
			else if(i==4){$this.addClass('dist-animation-logo2-animate');}
			else if(i==2){$this.addClass('dist-animation-logo3-animate');}
			else if(i==3){$this.addClass('dist-animation-logo5-animate');}
			  i++;
		  }, $delay3);
		});
	  }
	function isScrolledIntoView(elem) {
		var docViewTop = $(window).scrollTop();
		var docViewBottom = docViewTop + $(window).height();
		var elemTop = $(elem).offset().top;
		var elemBottom = elemTop + $(elem).height();
		return ((elemBottom-100 <= docViewBottom) && (elemTop >=docViewTop ));
	}
	if($(window).width()<=525){
	$("#mypulse_mobile>a").attr('href','http://onelink.to/vhzzxp');
	}
	$(window).scroll(function(){
        if(isScrolledIntoView('.retail_section') === true) {
      		aniamte_logos_retail();
        }
		else if(isScrolledIntoView('.restaurant_section') === true) {
      		aniamte_restaurant_smile();
        }
		else if (isScrolledIntoView('.distribution_section') === true) {
      		aniamte_dist_logo();
        }
    });
});

$(document).ready(function() {
	function aniamte_logos_retail(){
		var i=1;
		$delay = 0;
		var k=0;
		$('.retail-animation').each(function() {
		  	var $this = $(this);
			if(k==1){$delay= 600;}else if(k==2){$delay= 1500;}else if(k==3){$delay= 2100;}
			k++;
		  setTimeout(function() {
		  //$this.addClass('animate');
			if(i==1){$this.addClass('sales-today-text-animation');}
			else if(i==2){$this.addClass('sales-today-amount-animation')}
			else if(i==3){$this.addClass('grossmargin-text-animation')}
			else if(i==4){$this.addClass('grossmargin-amount-animation')}
			i++;
		  }, $delay);
		});
	  }
	  function aniamte_restaurant_smile(){
		var i=1;
		$delay2 = 0;
		  var j=0;
		$('.restaurant-smile').each(function() {
		  var $this = $(this);
			j++;
		  $delay2 += 200;if(j==6){$delay2 += 150;}
		  setTimeout(function() {
			//$this.addClass('animate');
			if(i==1){$this.addClass('restaurant-smile1-animation');}
			else if(i==2){$this.addClass('restaurant-smile2-animation');}
			else if(i==3){$this.addClass('restaurant-smile3-animation');}
			else if(i==4){$this.addClass('restaurant-smile4-animation');}
			else if(i==5){$this.addClass('restaurant-smile5-animation');}
			else if(i==6){$this.addClass('home-my-pulse-img-animation');}
			i++;
		  }, $delay2);
		});
	  }
	  function aniamte_dist_logo(){
		var i=1;
		$delay3 = 0;
		$('.dist-animation').each(function() {
		  var $this = $(this);
		  $delay3 += 250;
		  setTimeout(function() {
			//$this.addClass('animate');
			if(i==1){$this.addClass('dist-animation-logo1-animate');}
			else if(i==5){$this.addClass('dist-animation-logo4-animate');}
			else if(i==4){$this.addClass('dist-animation-logo2-animate');}
			else if(i==2){$this.addClass('dist-animation-logo3-animate');}
			else if(i==3){$this.addClass('dist-animation-logo5-animate');}
			  i++;
		  }, $delay3);
		});
	  }
	function isScrolledIntoView(elem) {
		var docViewTop = $(window).scrollTop();
		var docViewBottom = docViewTop + $(window).height();
		var elemTop = $(elem).offset().top;
		var elemBottom = elemTop + $(elem).height();
		return ((elemBottom-100 <= docViewBottom) && (elemTop >=docViewTop ));
	}
	if($(window).width()<=525){
	$("#mypulse_mobile>a").attr('href','http://onelink.to/vhzzxp');
	}
	$(window).scroll(function(){
        if(isScrolledIntoView('.retail_section') === true) {
      		aniamte_logos_retail();
        }
		else if(isScrolledIntoView('.restaurant_section') === true) {
      		aniamte_restaurant_smile();
        }
		else if (isScrolledIntoView('.distribution_section') === true) {
      		aniamte_dist_logo();
        }
    });
});
