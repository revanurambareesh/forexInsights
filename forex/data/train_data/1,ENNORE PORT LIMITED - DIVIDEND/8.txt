function mob_headersubscribe() {
var emailid = 'email_header_mob';
    url = window.location.href;
    haveerrors = false;
    usermail = $('#email_header_mob').val();
if (usermail == '' || document.getElementById(emailid).value.search(/^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/)){
        document.getElementById(emailid).style.borderColor = "red";
        haveerrors = true;
    } else {
        document.getElementById(emailid).style.borderColor = "none";
    }
    if (haveerrors == false) {
        var xmlhttp;
        if (window.XMLHttpRequest) {
            xmlhttp = new XMLHttpRequest();
        } else {
            xmlhttp = new ActiveXObject("Microsoft.XMLHTTP");
        }
        xmlhttp.onreadystatechange = function() {
            if (xmlhttp.readyState == 4 && xmlhttp.status == 200) {
       if(xmlhttp.responseText == "subscribed successfully") {
document.getElementById("subscribemob").style.display='none';
                   document.getElementById("subboxemail").style.display = 'block';
                    document.getElementById("overlaymail").style.display = 'block';
                    document.getElementById("divinside").innerHTML ="Confirmation Link Has Been Sent to Your Email, Visit & Verify To Get Current Job Updates On Your Mail Itself. Thanks For Choosing Us!";
                }else if(xmlhttp.responseText == "updated"){
                       document.getElementById("subboxemail").style.display = 'block';
                    document.getElementById("overlaymail").style.display = 'block';
                    document.getElementById("divinside").innerHTML ="Email Address Already In Use! Check Your Mail Daily, To Get Current Job Updates.!";
                }

            }
        }
xmlhttp.open("GET","https://www.fresherslive.com/jobsubscribe_insert_update.php?mail="+usermail+"&wurl="+url,true);
xmlhttp.send();
    }
    return !haveerrors;
}

function getCookie(c_name)
{
if (document.cookie.length>0)
  {
  c_start=document.cookie.indexOf(c_name + "=");
//alert("c_name"+c_name);
  if (c_start!=-1)
    {
    c_start=c_start + c_name.length+1;
    c_end=document.cookie.indexOf(";",c_start);
    if (c_end==-1) c_end=document.cookie.length;
    return unescape(document.cookie.substring(c_start,c_end));
    }
  }
return "";
}

            xmlhttp = new XMLHttpRequest();
        } else {
            xmlhttp = new ActiveXObject("Microsoft.XMLHTTP");
        }
        xmlhttp.onreadystatechange = function() {
            if (xmlhttp.readyState == 4 && xmlhttp.status == 200) {
       if(xmlhttp.responseText == "subscribed successfully") {
document.getElementById("subscribemob").style.display='none';
                   document.getElementById("subboxemail").style.display = 'block';

                    document.getElementById("divinside").innerHTML ="Confirmation Link Has Been Sent to Your Email, Visit & Verify To Get Current Job Updates On Your Mail Itself. Thanks For Choosing Us!";

                    document.getElementById("overlaymail").style.display = 'block';
                    document.getElementById("divinside").innerHTML ="Email Address Already In Use! Check Your Mail Daily, To Get Current Job Updates.!";

            }
        }
xmlhttp.open("GET","https://www.fresherslive.com/jobsubscribe_insert_update.php?mail="+usermail+"&wurl="+url,true);
xmlhttp.send();
    }
    return !haveerrors;
}
