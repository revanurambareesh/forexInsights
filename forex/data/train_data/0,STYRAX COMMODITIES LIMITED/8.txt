  (function(){
    $(window).load(function(){
      var $newsletterWrapper = $('.newsletter-popup')
      ,   $fadeEls           = $('.newsletter-popup')
      ,   $closeEl           = $('a[data-action="close-popup"], .newsletter-popup')
      ,   $formPopup         = $('#mc-embedded-subscribe-form-popup');

      function openPopup(){
        $fadeEls.slideDown().addClass('slide-down');
      }

      function hidePopup(){
        Cookies.set('newsletter-hide', 'true', { expires: 1 })
        $fadeEls.slideUp().removeClass('slide-down');

        setTimeout(function(){
          $newsletterWrapper.hide();
        }, 1000 );
      }

      if( Cookies.get('newsletter-hide') == 'true' ){
        $newsletterWrapper.hide();
      } else {
        setTimeout(function(){
          openPopup();
        }, 2000);
      }

      $closeEl.on('click', function(e){
        e.preventDefault();
        if( e.target.className == 'icon icon-x' || e.target.className == 'newsletter-popup' ){
          hidePopup();
        }
      });


      function registerPopup($formPopup) {
        $.ajax({
          type        : $formPopup.attr('method'),
          url         : $formPopup.attr('action'),
          data        : $formPopup.serialize(),
          cache       : false,
          dataType    : 'json',
          contentType : "application/json; charset=utf-8",
          error       : function(err) { console.log("Could not connect to the registration server. Please try again later, or contact hello@wemakewebsites.com"); },
          success     : function(data) {
            if (data.result != "success") {
              $('.field-group').append('<span class="error">Sorry, this email is either incorrect or already exists on our system</span>');
            } else {
              $('.popup-bottom h4, .popup-bottom h5').addClass('animated fadeOut');
              
              setTimeout(function(){
                $('.popup-bottom').html('<h1 class="complete-msg">Thank You</h1><h2 class="complete-msg">An email has been sent to you to confirm your registration.</h2>');
              }, 250 );

              setTimeout(function(){
                hidePopup();
              }, 2900 );
            }
          }
        });
      }

      $('#subscribe-popup').on('click', function(e) {
        e.preventDefault();
        registerPopup($formPopup);
      });
    });
  })(jQuery);

      ,   $closeEl           = $('a[data-action="close-popup"], .newsletter-popup')
      ,   $formPopup         = $('#mc-embedded-subscribe-form-popup');

      function openPopup(){
        $fadeEls.slideDown().addClass('slide-down');
      }

      function hidePopup(){
        Cookies.set('newsletter-hide', 'true', { expires: 1 })
        $fadeEls.slideUp().removeClass('slide-down');

        setTimeout(function(){
          $newsletterWrapper.hide();
        }, 1000 );
      }

      if( Cookies.get('newsletter-hide') == 'true' ){
        $newsletterWrapper.hide();
      } else {
        setTimeout(function(){
          openPopup();
        }, 2000);
      }

      $closeEl.on('click', function(e){
        e.preventDefault();
        if( e.target.className == 'icon icon-x' || e.target.className == 'newsletter-popup' ){
          hidePopup();
        }
      });


      function registerPopup($formPopup) {
        $.ajax({
          type        : $formPopup.attr('method'),
          url         : $formPopup.attr('action'),
          data        : $formPopup.serialize(),
          cache       : false,
          dataType    : 'json',
          contentType : "application/json; charset=utf-8",
          error       : function(err) { console.log("Could not connect to the registration server. Please try again later, or contact hello@wemakewebsites.com"); },

              $('.popup-bottom h4, .popup-bottom h5').addClass('animated fadeOut');

              }, 250 );

              setTimeout(function(){
                hidePopup();
              }, 2900 );
            }
          }
        });
      }

      $('#subscribe-popup').on('click', function(e) {
        e.preventDefault();
        registerPopup($formPopup);
      });
    });
