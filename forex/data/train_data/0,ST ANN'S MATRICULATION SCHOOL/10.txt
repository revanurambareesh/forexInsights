function setCookie(name,value,days) {
    if (days) {
        date.setTime(date.getTime()+(days*24*60*60*1000));
        var expires = "; expires="+date.toGMTString();
    }
    else var expires = "";
    document.cookie = name+"="+value+expires+"; path=/";
}


function getCookie(name) {
    var nameEQ = name + "=";
    var ca = document.cookie.split(';');
    for(var i=0;i < ca.length;i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
    }
    return null;
}

function deleteCookie(name) {
    setCookie(name,"",-1);
}

function is_whitelisted_url(){
    path = window.location.pathname;
    path_parts = path.split('/');
    //whitelist only the school member area
    if((path_parts) && (path_parts.length > 1) && (path_parts[1] == 'schools')){
        return true;
    } else {
        return false;
    }
}

    //do nothing if it's a whitelisted URL
    if (is_whitelisted_url()) {
        return;
    }
    
    //get the cookie
    c = getCookie('interstitial_ad'); 
    //c is for cookie!
    c_count = 1;
    c_date = '';
  
    day = 1000*60*60*24;
    now = date.getTime();
    yesterday = now - day;
  
    if (!c) {
        c_date = date.getTime();
    } else {
        //parse the cookie value
        cookie_fields = c.split('|');
        c_date = cookie_fields[1];
        if (c_date > yesterday) {
            //if the date is within 24 hours, increment the count and leave the date as-is
            c_count = parseInt(cookie_fields[0]) + 1;
        } else {
            //if the date is older than 24 hours, reset the count and set the date to now
            c_count = 1;
            c_date = now;
        }
    }
  
    //save the new values
    setCookie('interstitial_ad', c_count + '|' + c_date);
    var width= window.innerWidth || (window.document.documentElement.clientWidth || window.document.body.clientWidth);

function AjaxSearchmobile_header_search_form(){

        return $( "<li></li>" )
            .data( "item.autocomplete", item )
            .append( $( "<a></a>" ).html( item.label ) )
            .appendTo( ul );
    };
    $("#mobile-header-search-form .search-input").autocomplete({

                dataType: "json",

                    if (data) {
                        response(
                            $.map(data, function (item) {
                                return{
                                    label: item.label,
                                    link: item.link,
                                    value: item.text
                                }
                            })
                        );
                    } else {
                        response(false, false);
                    }
                }
            })
        },
        select: function (event, ui) {
            window.location=ui.item.link;
            return false;
        },
        change: function (event, ui) {
            if(ui.item){
                $("#mobile-header-search-form .search-input").val(ui.item.value);
                window.location=ui.item.link;
            }
        },
        open: function(){
            $(".ui-menu").width("auto");
        }
    }).data( "autocomplete" )._renderItem = function( ul, item ) {
        return $( "<li></li>" )
            .data( "item.autocomplete", item )
            .append( "<a>"+ item.label + "</a>" )
            .appendTo( ul );
    };

function setCookie(name,value,days) {
    if (days) {
        date.setTime(date.getTime()+(days*24*60*60*1000));
        var expires = "; expires="+date.toGMTString();
    }
    else var expires = "";
    document.cookie = name+"="+value+expires+"; path=/";
}


function getCookie(name) {
    var nameEQ = name + "=";
    var ca = document.cookie.split(';');
    for(var i=0;i < ca.length;i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
    }
    return null;
}

function deleteCookie(name) {
    setCookie(name,"",-1);
}

function is_whitelisted_url(){
    path = window.location.pathname;
    path_parts = path.split('/');
    //whitelist only the school member area
    if((path_parts) && (path_parts.length > 1) && (path_parts[1] == 'schools')){
        return true;
    } else {
        return false;
    }
}

function ad_overlay_main(){ 
    //do nothing if it's a whitelisted URL
    if (is_whitelisted_url()) {
        return;
    }
    
    //get the cookie
    c = getCookie('interstitial_ad'); 
    //c is for cookie!
    c_count = 1;
    c_date = '';
  
    day = 1000*60*60*24;
    now = date.getTime();
    yesterday = now - day;
  
    if (!c) {
        c_date = date.getTime();
    } else {
        //parse the cookie value
        cookie_fields = c.split('|');
        c_date = cookie_fields[1];
        if (c_date > yesterday) {
            //if the date is within 24 hours, increment the count and leave the date as-is
            c_count = parseInt(cookie_fields[0]) + 1;
        } else {
            //if the date is older than 24 hours, reset the count and set the date to now
            c_count = 1;
            c_date = now;
        }
    }
  
    //save the new values
    setCookie('interstitial_ad', c_count + '|' + c_date);

function AjaxSearchcse_search_box_1(){
    $[ "ui" ][ "autocomplete" ].prototype["_renderItem"] = function( ul, item) {
        return $( "<li></li>" )
            .data( "item.autocomplete", item )
            .append( $( "<a></a>" ).html( item.label ) )
            .appendTo( ul );
    };
    $("#cse-search-box-1 .search-input").autocomplete({
        source: function (request, response) {
            var search_string = ($("#search_bar_moblie").is(":visible")) ? $("#cse-search-box-1 .search-input").eq(1).val() : $("#cse-search-box-1 .search-input").val();
            $.ajax({
                url: "http://dynamic.schoolreviewservices.com/search/bsr/"+search_string,
                type: "GET",
                dataType: "json",
                success: function (data) {
                    if (data) {
                        response(
                            $.map(data, function (item) {
                                return{
                                    label: item.label,
                                    link: item.link,
                                    value: item.text
                                }
                            })
                        );
                    } else {
                        response(false, false);
                    }
                }
            })
        },
        select: function (event, ui) {
            window.location=ui.item.link;
            return false;
        },
        change: function (event, ui) {
            if(ui.item){
                $("#cse-search-box-1 .search-input").val(ui.item.value);
                window.location=ui.item.link;
            }
        },
        open: function(){
            $(".ui-menu").width("auto");
        }
    }).data( "autocomplete" )._renderItem = function( ul, item ) {
        return $( "<li></li>" )
            .data( "item.autocomplete", item )
            .append( "<a>"+ item.label + "</a>" )
            .appendTo( ul );
    };
}
 
 
function SearchboxAction(){
    $(".cse-search-box button[type=submit]").removeAttr("disabled");
    $(".cse-search-box").each(function() {
        var f = $(this).get(0);
        var q = $(this).find("input[name=q]");
        var input_elements = q.get(0);
        var fomr_id = $(this).attr("id");
        var input_id = q.attr("id");
        var x_delete = $(this).find("span[class=x_delete]");
        var n = navigator;
        var l = location;
        var su = function() {
                var siteurl_input=$(this).find("input[name=siteurl]");
                var v = document.location.toString();
                var existingSiteurl = /(?:[?&]siteurl=)([^&#]*)/.exec(v);
                if (existingSiteurl) {
                    v = decodeURI(existingSiteurl[1])
                }
                var delimIndex = v.indexOf("://");
                if (delimIndex >= 0) {
                    v = v.substring(delimIndex + "://".length, v.length)
                }
                siteurl_input.val(v);
            };
        if (window.history.navigationMode) {
            window.history.navigationMode = "compatible"
        }
        var b = "#fff";
        q.css({"background":b,"outline":"none"});
        q.attr("readonly",false);
        q.focus(function() {
            if (q.val() == "Enter school name, location or keyword") {
                q.css({"color":"#000"});
                q.val("");
            }
        });
        q.blur(function() {
            if (q.val() == "") {
                q.css({"color":"#a9a9a9"});
                q.val("Enter school name, location or keyword");
            }
        });
        q.keyup(function() {
            x_delete.fadeIn();
            if ($.trim(q.val()) == "") {
                x_delete.fadeOut();
            }
        });
        x_delete.click(function() {
            q.css({"color":"#a9a9a9"});
            q.val("Enter school name, location or keyword");
            x_delete.hide();
        });
        $(this).submit(function() {
            if (q.val() == "Enter school name, location or keyword") {
                q.val("");
            }
        });
    });
}

function ShowMobileNavBar(){

        function(){

            $("#m-nav-bar").width(parseInt(m_bar_width * container_width)).show().css("float","right").height($("#page-container").height());

            $("#page-container").width("100%").css("float","none");
            $("#search_footer_div").width("100%").css("float","none");

function HoverOnSubcategoryMenu(){
    $(".navigation li.dropdown4").mouseenter(function() {
        var p_str = $(".navigation li.dropdown4").width();
        var m_str = $("#down-list4").width();
        var off_set = $(".navigation li.dropdown4").offset().left - m_str + p_str;
        t4=setTimeout(function(){
            $("#down-list4").show().offset({left:off_set});
            $(".navigation li.dropdown4").addClass("navigation_hover");
        },300)
    });
    $("#down-list4").mouseenter(function() {
        var p_str = $(".navigation li.dropdown4").width();
        var m_str = $("#down-list4").width();
        var off_set = $(".navigation li.dropdown4").offset().left - m_str + p_str;
        $("#down-list4").show().offset({left:off_set});
        $(".navigation li.dropdown4").addClass("navigation_hover");
    });
    $(".navigation li.dropdown4, #down-list4").mouseleave(function(){
        clearTimeout(t4);
            $("#down-list4").hide();
            $(".navigation li.dropdown4").removeClass("navigation_hover");
    });

 function setCircle(){
    $('.mask').each(function(){
        var value = $(this).children('span').text();
        if( value <= 50 ){
            $(this).siblings('.pie_right').css('transform','rotate('+(value*3.6)+'deg)');
