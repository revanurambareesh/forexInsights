function validateReg()
{
	
	if(document.FormReg.txtName.value=="")
	{
		alert("please enter name!");
		document.FormReg.txtName.focus();
		return false;
	}
	
	if(document.FormReg.txtEmail.value=="")
	{
		alert("please enter Email Address!");
		document.FormReg.txtEmail.focus();
		return false;
	}
	
	if (verify_email("document.FormReg","txtEmail") == false)
	{
	alert("The email address entered seems to be incorrect. Please fill in a correct email");
	document.FormReg.txtEmail.focus();
	return false;
	}
	
	if(document.FormReg.txtMobNo.value=="")
	{
		alert("please enter Mobile No!");
		document.FormReg.txtMobNo.focus();
		return false;
	}
	
	if(isNaN(document.FormReg.txtMobNo.value))
	{
		alert("Please enter integer value!")
		document.FormReg.txtMobNo.focus();
		return false;
	}
	
	if(document.FormReg.txtMobNo.value.length <10 || document.FormReg.txtMobNo.value.length > 15)
	{
		alert("Please enter 10 to 15 digit!")
		document.FormReg.txtMobNo.focus();
		return false;
	}
	if (checkForEmptyField("document.FormReg",essential_fields_array) == false)
		return false;
     return true
}

function validateReg()
{
	
	if(document.FormReg.txtName.value=="")
	{
		alert("please enter name!");
		document.FormReg.txtName.focus();
		return false;
	}
	
	if(document.FormReg.txtEmail.value=="")
	{
		alert("please enter Email Address!");
		document.FormReg.txtEmail.focus();
		return false;
	}
	
	if (verify_email("document.FormReg","txtEmail") == false)
	{
	alert("The email address entered seems to be incorrect. Please fill in a correct email");
	document.FormReg.txtEmail.focus();
	return false;
	}
	
	if(document.FormReg.txtMobNo.value=="")
	{
		alert("please enter Mobile No!");
		document.FormReg.txtMobNo.focus();
		return false;
	}
	
	if(isNaN(document.FormReg.txtMobNo.value))
	{
		alert("Please enter integer value!")
		document.FormReg.txtMobNo.focus();
		return false;
	}
	
	if(document.FormReg.txtMobNo.value.length <10 || document.FormReg.txtMobNo.value.length > 15)
	{
		alert("Please enter 10 to 15 digit!")
		document.FormReg.txtMobNo.focus();
		return false;
	}
	if (checkForEmptyField("document.FormReg",essential_fields_array) == false)
		return false;
     return true
}
