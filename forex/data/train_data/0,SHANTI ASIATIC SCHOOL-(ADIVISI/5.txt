 * This function creates a bean in backend in first request. in second request it populates the data
 * @param formId
 * @param formUrl
 */
function submitFormUsingGet(formId, formUrl){
	var jQueryFormId = "#" + formId ;
	AJAX_GET_JSON_DATA("/createjobsearchbean", $("#" + formId ).serialize(), function(data){
		$.each(data, function(key, value) { 
			if(key == 'success' && value == 'success'){
				//now send get request
				window.location.href = formUrl;
 			}   
		});
	});
}

function makeAutoCompleter(source_id,purl,index) {
	is_selected = false;
	$('#' + source_id).autocomplete(
			{
				source : function(request, response) {
					$.ajax({
						url :  purl + "?matchString="
								+ request.term,
						dataType : "json",
						success : function(data) {
							response($.map(data, function(v, i) {
								v = v.replace('<b>',"").replace('</b>',"");
								return {
									label : i,
									value : v
								};
							}));
						}
					});
				},
				focus : function(event, ui) {
						$('#'+source_id).val(ui.item.value);
				},
				minLength : 3,
				select : function(event, ui) {
					$('#'+source_id).val(ui.item.value);
					index++;
					return false;
				}
			}).data("ui-autocomplete")._renderItem = function(ul, item) {
		return $("<li>").append("<a>" + item.value + "</a>").appendTo(ul);
	};
}
