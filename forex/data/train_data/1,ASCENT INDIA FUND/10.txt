function submitSubscribe(emailfield,subscribebtn,div_thanks,div_errmsg)
{
	var email =  document.getElementById(emailfield).value;
	var sub_captcha = "";
	var section = "interview";
	var article_url = "http://business.rediff.com/interview/2010/feb/10/inter-foreign-investors-prefer-independent-managers.htm";
	var subreq;
	if(section == "business"){section = "money" ;}
	
	if(!validEmail(email))
	{
		alert("Please enter a valid email");
		document.getElementById(emailfield).focus();
		return false;
	}
	if(emailfield == "sub_email_in")
	{
		sub_fillTacking('email_topbox',emailfield);
	}
	else
	{
		sub_fillTacking('email',emailfield); 
	}

	var url= cdn_domain+"/newsletter";
	var postdata = "email="+email+"&captcha="+sub_captcha+"&section="+section+"&article_url="+encodeURIComponent(article_url)+"&country=india&output=xml";
	if (window.XMLHttpRequest)
	{
		subreq = new XMLHttpRequest();
		subreq.onreadystatechange = statusstateChanged;
		subreq.open("POST", url, true);
		subreq.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
		subreq.setRequestHeader("Content-Length", postdata.length.toString());
		subreq.setRequestHeader("Connection", "Close");
		subreq.send(postdata);
	}
	else if (window.ActiveXObject)
	{
		subreq = new ActiveXObject("Microsoft.XMLHTTP");
		if (subreq)
		{
			subreq.onreadystatechange = statusstateChanged;
			subreq.open("POST", url, true);
			subreq.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
			subreq.setRequestHeader("Content-Length", postdata.length);
			subreq.setRequestHeader("Connection", "Close");
			subreq.send(postdata);
		}
	}

	function statusstateChanged()
	{
		if (subreq.readyState==4 || subreq.readyState=="complete")
		{
			if(subreq.status == 200)
			{
				var resp_status = (subreq.responseXML.getElementsByTagName("status")[0].firstChild)?subreq.responseXML.getElementsByTagName("status")[0].firstChild.nodeValue:0;
				var resp_message = (subreq.responseXML.getElementsByTagName("message")[0].firstChild)?subreq.responseXML.getElementsByTagName("message")[0].firstChild.nodeValue:0;
				if(resp_status == "Success")
				{
					if(document.getElementById(div_thanks)){document.getElementById(div_thanks).innerHTML = "<div style='padding:0 0 5px 0;overflow:hidden;border:solid 1px #ccc;background:#fff;text-align:center;'><div title='Close' style='float:right;color:#0066bc;font:bold 16px arial;' class='floatR bold curhand f14' onclick='closeSubscripBox();'>x &nbsp;</div><span class='clear'></span><p class='f14 bold alignC'>Thank you for signing up for the Rediff NewsLetter!</p></div>" ;}
					if(document.getElementById('metric_NLiframe'))
					{
						if(emailfield == "sub_email_in")
						{
							document.getElementById('metric_NLiframe').src = "http://metric.ind.rediff.com/www.rediff.com/nlsubscribe/thanks_topbox?rkey="+Math.random();
						}
						else
						{
							document.getElementById('metric_NLiframe').src = "http://metric.ind.rediff.com/www.rediff.com/nlsubscribe/thanks?rkey="+Math.random();
						}
					}
				}
				else
				{
					document.getElementById(div_errmsg).innerHTML  = "<p class=red>"+resp_message+"</p>"; 
					document.getElementById(div_errmsg).style.display = "block";
					if(document.getElementById(subscribebtn)){document.getElementById(subscribebtn).disabled = false;}
				}
			}
		}
		if (subreq.readyState == 1)
		{
			if(document.getElementById(subscribebtn)){document.getElementById(subscribebtn).disabled = true;}
		}
	}
	return false;
}
