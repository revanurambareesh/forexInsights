function flex_banner_init(d){
    var sec = $(d.vars.sectionsContainer);
    $("ul li",sec).each(function(i){
        if(i == 0) $(this).addClass("current");   
        var id = $(this).attr("id");
        d.slides.filter(function(j){return $(this).attr("rel") == id;}).appendTo(d.container);       
    });
    $("ul li a",sec).live("click",function(){
        var li = $(this).parent("li");
        var ix = d.slides.filter("[rel="+li.attr("id")+"]").first().index();
        ix = ix - d.cloneOffset;
        d.flexAnimate(ix,d.vars.pauseOnAction);          
    });
}
function flex_banner_animation_starts(d){ 
    var sec = $(d.vars.sectionsContainer);
    var ix = d.animatingTo+d.cloneOffset;
    var id = d.newSlides.eq(ix).attr("rel");    
    $("ul li",sec).removeClass("current").filter("#"+id).addClass("current");
}
