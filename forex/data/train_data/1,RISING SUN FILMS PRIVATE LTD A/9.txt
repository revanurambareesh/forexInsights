	if(userInfo !=null && userInfo!= 'undefined'){

		if(userType !=null && userType!= 'undefined' && userType.s != 'undefined' && userType.s == 't'){
			isPaidUser="Y";
		}else{
			isPaidUser="N";
		}
	}
	
	
	if(author_id != ''){
		if(confirm("Are you sure to unfollow this columnist!")==true){
	        $.ajax({
	            url : '/request-handler/article/unmap-user-columnist',
	            type : 'post',
	            data: {
					'authorId'	: author_id,'isPaidUser':isPaidUser
				},
	            success : function(msg){
	                if(msg == 'success'){
	                	location.reload(); return false();
	                }else{
	                    message = 'Please try again';
	                    alert(message); return false();
	                }
	            }
	        });
		}
    }else{
       alert("Invalid columnist request!");
       return false();
    }

}
