        function divexpandcollapse(divname, urlname, urlname1, type) {


            if ((urlname.trim() == "#") || (urlname.trim() == "undefined")) {
                var hidediv;
                var imghide;
                var rowcount = $('#lblcount').text();
                var div = document.getElementById(divname);
                var img = document.getElementById('img' + divname);
                var duplicatediv = "div";
                var applydiv;
                var e = document.getElementById("drp_OrderId");
                for (var i = 0; i < rowcount; i++) {
                    applydiv = duplicatediv.concat(e.options[i].value);
                    if (applydiv.toString() == divname.toString()) {
                        if (div.style.display == "none") {
                            div.style.display = "block";
                            img.src = "minus1.png";
                            img.style.width = "29";
                            $(div).hide().slideDown('slow');
                            $('#footer-box').hide().slideDown('slow');
                        }
                        else {
                            $(div).slideToggle();
                            img.src = "plus1.png";
                        }
                    }
                    else {
                        hidediv = document.getElementById(applydiv);
                        if (hidediv == null) { }
                        else {
                            imghide = document.getElementById('img' + applydiv);
                            hidediv.style.display = "none";
                            imghide.src = "plus1.png";
                            $(hidediv).slideUp(2000);
                        }
                    }
                }
            }
            else {
                if (type.toString() == "_blank") {
                    window.open(urlname1);
                }
                else
                { window.location = urlname1; }
            }
            footerDown1();
        }

        function filterOverlaps(collection1, collection2) {
            //// alert(collection1);
            ///alert(collection2);
            var dims1 = getDims(collection1),
            dims2 = !collection2 ? dims1 : getDims(collection2),
            stack = [],
            index1 = 0,
            index2 = 0,
            length1 = dims1.length,
            length2 = !collection2 ? dims1.length : dims2.length;

            if (!collection2) { collection2 = collection1; }

            for (; index1 < length1; index1++) {
                for (index2 = 0; index2 < length2; index2++) {
                    if (collection1[index1] === collection2[index2]) {
                        continue;
                    }
                    else if (checkOverlap(dims1[index1], dims2[index2])) {
                        return true;
                        // stack.push((length1 > length2) ?collection1[index1]:collection2[index2]);
                    }
                }
            }

            //alert($.unique(stack));
        }

        function getDims(elems) {

            var dims = [], i = 0, offset, elem;

            while ((elem = elems[i++])) {
                offset = $(elem).offset();
                dims.push([(offset.top - $(window).scrollTop()), offset.left, elem.offsetWidth, elem.offsetHeight]);
            }

            return dims;
        }
